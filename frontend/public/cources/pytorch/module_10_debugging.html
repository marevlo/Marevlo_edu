SECTION 01

Detecting Vanishing & Exploding Gradients

The Problem

  • Vanishing gradients: gradients become extremely small in early layers → model stops learning.
  • Exploding gradients: gradients grow exponentially → loss becomes NaN.
Gradient Norm Monitor
Backprop through a deep network and inspect the gradient norm at every layer. Switch initialisation strategy and activation to see vanishing, exploding, or healthy gradients.

import torch
import torch.nn as nn
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

torch.manual_seed(42)

def monitor_gradients(model: nn.Module, verbose: bool = True) -> dict:
    """
    After loss.backward(), inspect gradient statistics for each layer.
    """
    stats = {}
    for name, param in model.named_parameters():
        if param.grad is not None:
            g = param.grad.data
            stats[name] = {
                'mean': g.mean().item(),
                'std':  g.std().item(),
                'max':  g.abs().max().item(),
                'norm': g.norm().item()
            }
            if verbose:
                status = ''
                if g.norm().item() < 1e-7: status = ' ← VANISHING'
                if g.norm().item() > 100:   status = ' ← EXPLODING'
                print(f"  {name:30s} norm={g.norm().item():.2e}  "
                      f"max={g.abs().max().item():.2e}{status}")
    return stats


# Demo: deep network with no initialisation care (can vanish)
class DeepNet(nn.Module):
    def __init__(self, depth=10, bad_init=False):
        super().__init__()
        layers = []
        for _ in range(depth):
            l = nn.Linear(32, 32)
            if bad_init:
                nn.init.uniform_(l.weight, -5, 5)   # Terrible init → exploding
            layers.extend([l, nn.Tanh()])
        self.net = nn.Sequential(*layers, nn.Linear(32, 1))

    def forward(self, x):
        return self.net(x)


# Good init

loss.backward()

SECTION 02

Activation & Gradient Histograms

Visualising the distribution of activations and gradients helps you see:

  • Dead neurons (ReLU activations always zero)
  • Saturation (tanh/sigmoid always at ±1)
  • Vanishing/exploding gradients
Per-Layer Activation Histograms
Forward-pass histograms across 6 hidden layers. Healthy networks show wide, roughly Gaussian distributions. Dead neurons, saturation, and collapse jump out immediately.

class ActivationMonitor:
    """Hook-based activation and gradient recorder."""

    def __init__(self):
        self.activations = {}
        self.gradients   = {}
        self.hooks       = []

    def register(self, model: nn.Module):
        for name, layer in model.named_modules():
            if isinstance(layer, (nn.Linear, nn.Conv2d)):
                self.hooks.append(
                    layer.register_forward_hook(self._make_act_hook(name))
                )
                self.hooks.append(
                    layer.register_backward_hook(self._make_grad_hook(name))
                )

    def _make_act_hook(self, name):
        def hook(module, input, output):
            self.activations[name] = output.detach().cpu()
        return hook

    def _make_grad_hook(self, name):
        def hook(module, grad_input, grad_output):
            self.gradients[name] = grad_output[0].detach().cpu() if grad_output[0] is not None else None
        return hook

    def remove(self):
        for h in self.hooks: h.remove()

    def summary(self):
        print("Layer activation statistics:")
        for name, act in self.activations.items():
            dead = (act == 0).float().mean().item()
            print(f"  {name:20s}: mean={act.mean():.3f}  std={act.std():.3f}  "
                  f"dead_ratio={dead:.2f}")


monitor = ActivationMonitor()
model   = DeepNet(depth=4)
monitor.register(model)

x    = torch.randn(32, 32)
out  = model(x)
loss = out.sum()
loss.backward()
monitor.summary()
monitor.remove()

Learning Rate Finder

The LR finder sweeps the learning rate from very small to very large over a few iterations, recording the loss. The optimal LR is just before the loss starts climbing.

Learning Rate Finder
Sweep LR from 10⁻⁷ to 10¹ on a log scale. The model trains for one step per LR. The optimal LR is at the steepest downward slope — just before the loss explodes.

def lr_finder(model: nn.Module, X: torch.Tensor, y: torch.Tensor,
              criterion, start_lr: float = 1e-7, end_lr: float = 10.0,
              n_steps: int = 100) -> tuple:
    """
    Sweeps LR from start_lr to end_lr over n_steps.
    Returns: (lrs, losses)
    """
    # Save original state
    orig_state = {k: v.clone() for k, v in model.state_dict().items()}

    opt = torch.optim.SGD(model.parameters(), lr=start_lr)
    lrs, losses = [], []
    best_loss   = float('inf')
    lr          = start_lr

    multiplier = (end_lr / start_lr) ** (1.0 / n_steps)

    for step in range(n_steps):
        opt.zero_grad()
        loss = criterion(model(X), y)
        loss.backward()
        opt.step()

        lrs.append(lr)
        losses.append(loss.item())

        if loss.item() < best_loss:
            best_loss = loss.item()
        if loss.item() > 4 * best_loss:       # Stop if loss explodes
            break

        lr *= multiplier
        for pg in opt.param_groups:
            pg['lr'] = lr

    # Restore original state
    model.load_state_dict(orig_state)
    return lrs, losses


# Run it
X_lr = torch.randn(128, 10)
y_lr = torch.randint(0, 3, (128,))

model_lr = nn.Sequential(
    nn.Linear(10, 64), nn.ReLU(),
    nn.Linear(64, 32), nn.ReLU(),
    nn.Linear(32, 3)
)

lrs, losses = lr_finder(model_lr, X_lr, y_lr, nn.CrossEntropyLoss())

# Find suggested LR (steepest gradient)
losses_arr = np.array(losses)
lrs_arr    = np.array(lrs)
grad       = np.gradient(losses_arr)
best_idx   = np.argmin(grad[:-5])   # Avoid end instability
print(f"LR range explored: {lrs_arr[0]:.2e} → {lrs_arr[-1]:.2e}")
print(f"Suggested LR: {lrs_arr[best_idx]:.2e}")

Sanity Checks

Overfit a Tiny Batch — The First Test You Should Always Run

"Overfit a Tiny Batch" Sanity Check
A correctly-implemented model should be able to drive the loss to ≈0 on just 4 samples. If it can't, something is fundamentally wrong — and you should fix it before scaling up.

def sanity_check_overfit(model: nn.Module, n_samples: int = 4,
                          n_steps: int = 200, lr: float = 1e-2) -> bool:
    """
    A model should be able to overfit to a tiny batch.
    If it can't, something is fundamentally wrong.
    """
    X = torch.randn(n_samples, 10)
    y = torch.randint(0, 3, (n_samples,))

    opt  = torch.optim.Adam(model.parameters(), lr=lr)
    crit = nn.CrossEntropyLoss()

    for step in range(n_steps):
        opt.zero_grad()
        loss = crit(model(X), y)
        loss.backward()
        opt.step()

    final_loss = loss.item()
    passed     = final_loss < 0.05
    print(f"Overfit test: final_loss={final_loss:.5f}  {'PASS ✓' if passed else 'FAIL ✗'}")
    return passed


model_sanity = nn.Sequential(nn.Linear(10, 64), nn.ReLU(), nn.Linear(64, 3))
sanity_check_overfit(model_sanity)

# === BROKEN MODEL — DO NOT TRAIN YET ===
# There are 5 bugs in this code. Find them before reading the solutions.

class BrokenModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(10, 64)
        self.fc2 = nn.Linear(64, 32)
        self.fc3 = nn.Linear(32, 3)
        self.dropout = nn.Dropout(0.5)
        # BUG 1: Using plain Python list — parameters won't be tracked
        self.extra_layers = [nn.Linear(64, 64), nn.Linear(64, 64)]

    def forward(self, x):
        x = torch.sigmoid(self.fc1(x))          # BUG 2: Sigmoid → vanishing gradients in deep networks
        x = self.fc2(x)
        x = torch.relu(x)
        x = self.dropout(x)
        x = self.fc3(x)
        x = torch.softmax(x, dim=1)             # BUG 3: Softmax before CrossEntropyLoss = numerical issues
        return x


def broken_training():
    model = BrokenModel()
    X = torch.randn(32, 10)
    y = torch.randint(0, 3, (32,))

    # BUG 4: Using Adam with default weight_decay=0 (no regularisation)
    # Should use AdamW with weight_decay
    opt = torch.optim.Adam(model.parameters(), lr=0.1)    # BUG 5: LR way too high

    criterion = nn.CrossEntropyLoss()

    for step in range(5):
        output = model(X)
        loss   = criterion(output, y)
        loss.backward()   # BUG: missing optimizer.zero_grad()!
        opt.step()
        print(f"Step {step}: loss={loss.item():.4f}")

# === FIXED MODEL ===
class FixedModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(10, 64)
        self.fc2 = nn.Linear(64, 32)
        self.fc3 = nn.Linear(32, 3)
        self.dropout = nn.Dropout(0.3)
        # FIX 1: Use nn.ModuleList
        self.extra_layers = nn.ModuleList([nn.Linear(64, 64), nn.Linear(64, 64)])

    def forward(self, x):
        # FIX 2: Use ReLU, not Sigmoid
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.dropout(x)
        # FIX 3: Return raw logits — CrossEntropyLoss handles softmax internally
        return self.fc3(x)


def fixed_training():
    model = FixedModel()
    X = torch.randn(32, 10)
    y = torch.randint(0, 3, (32,))

    # FIX 4 + 5: AdamW with sensible LR and weight decay
    opt = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-2)
    criterion = nn.CrossEntropyLoss(label_smoothing=0.05)

    for step in range(5):
        opt.zero_grad()   # FIX: zero gradients before backward
        loss = criterion(model(X), y)
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        opt.step()
        print(f"Step {step}: loss={loss.item():.4f}")


print("\n=== FIXED version ===")
