TorchScript

python
import torch
import torch.nn as nn

torch.manual_seed(42)

class ProductionModel(nn.Module):
    def __init__(self, in_dim: int = 10, n_classes: int = 3):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, 64), nn.LayerNorm(64), nn.ReLU(),
            nn.Linear(64, 32), nn.ReLU(),
            nn.Linear(32, n_classes)
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)


model = ProductionModel()
example_input = torch.randn(1, 10)

# Method 1: torch.jit.trace — fastest, for static graphs (no dynamic control flow)
traced = torch.jit.trace(model, example_input)
torch.jit.save(traced, "model_traced.pt")
print("Traced model saved.")

# Method 2: torch.jit.script — handles dynamic control flow
# Note: @torch.jit.export can be added to expose extra methods in script mode,
# but trace and script cannot be combined on the same class instance.
scripted = torch.jit.script(model)
torch.jit.save(scripted, "model_scripted.pt")
print("Scripted model saved.")

# Load and verify
loaded = torch.jit.load("model_scripted.pt")
loaded.eval()

x_test = torch.randn(4, 10)
with torch.no_grad():
    out   = loaded(x_test)
    proba = torch.softmax(out, dim=1)

print(f"Inference output shape: {out.shape}")
print(f"Probabilities sum to 1: {proba.sum(dim=1).round(decimals=4)}")
Traced model saved. Scripted model saved. Inference output shape: torch.Size([4, 3]) Probabilities sum to 1: tensor([1., 1., 1., 1.])

SECTION 02

Serving Patterns: Batch vs Realtime

python
# Batch inference — process many samples offline
def batch_inference(model_path: str, data: torch.Tensor,
                    batch_size: int = 256) -> torch.Tensor:
    """Load a scripted model and run inference in batches."""
    model = torch.jit.load(model_path)
    model.eval()

    results = []
    for i in range(0, len(data), batch_size):
        batch = data[i:i+batch_size]
        with torch.no_grad():
            logits = model(batch)                          # forward() — always available
            probs  = torch.softmax(logits, dim=1)         # apply softmax here
        results.append(probs)

    return torch.cat(results, dim=0)


# Test
test_data = torch.randn(1000, 10)
probs     = batch_inference("model_scripted.pt", test_data)
print(f"Batch inference: {len(test_data)} samples → {probs.shape}")
Batch inference: 1000 samples → torch.Size([1000, 3])

# This is the complete FastAPI service — save as inference_service.py
# Run with: uvicorn inference_service:app --host 0.0.0.0 --port 8000

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import torch
import time
import logging
from typing import List

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

app = FastAPI(title="PyTorch Model Inference Service")


# ─── Model Loading (once at startup) ───────────────────────────────────────
class ModelRegistry:
    def __init__(self):
        self.models = {}
        self.active = None

    def load(self, name: str, path: str):
        model = torch.jit.load(path)
        model.eval()
        self.models[name] = model
        logger.info(f"Loaded model: {name}")

    def predict(self, name: str, x: torch.Tensor) -> torch.Tensor:
        if name not in self.models:
            raise ValueError(f"Model {name} not loaded")
        with torch.no_grad():
            logits = self.models[name](x)
            return torch.softmax(logits, dim=1)


registry = ModelRegistry()

@app.on_event("startup")
def startup():
    registry.load("v1", "model_scripted.pt")
    registry.active = "v1"


# ─── Request / Response schemas ────────────────────────────────────────────
class PredictRequest(BaseModel):
    features:   List[List[float]]          # Batch of feature vectors
    model_name: str = "v1"


class PredictResponse(BaseModel):
    predictions:   List[int]
    probabilities: List[List[float]]
    model_name:    str
    latency_ms:    float


# ─── Endpoints ────────────────────────────────────────────────────────────
@app.post("/predict", response_model=PredictResponse)
def predict(request: PredictRequest):
    start = time.perf_counter()
    try:
        x       = torch.tensor(request.features, dtype=torch.float32)
        probs   = registry.predict(request.model_name, x)
        preds   = probs.argmax(dim=1).tolist()
        latency = (time.perf_counter() - start) * 1000

        logger.info(f"Predicted {len(preds)} samples in {latency:.2f}ms")

        return PredictResponse(
            predictions=preds,
            probabilities=probs.tolist(),
            model_name=request.model_name,
            latency_ms=latency
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))


@app.get("/health")
def health():
    return {"status": "ok", "active_model": registry.active}


@app.post("/models/{name}/activate")
def activate_model(name: str):
    """A/B testing or rollback: switch the active model."""
    if name not in registry.models:
        raise HTTPException(status_code=404, detail="Model not found")
    registry.active = name
    return {"activated": name}

Monitoring: Latency, Drift, Failure Patterns

Latency Monitoring

Latency Percentile Dashboard
200 inference requests with ~2% slow outliers. Drag the outlier rate and SLA threshold — watch which percentiles flip into alert state. Means lie; tails are what matter.
Outlier rate 2%
Outlier multiplier 10×
SLA p99 threshold 30 ms
python
import time
import statistics

class LatencyMonitor:
    """Track inference latency with rolling statistics."""

    def __init__(self, window: int = 1000):
        self.window    = window
        self.latencies = []

    def record(self, latency_ms: float):
        self.latencies.append(latency_ms)
        if len(self.latencies) > self.window:
            self.latencies.pop(0)

    def stats(self) -> dict:
        if not self.latencies:
            return {}
        return {
            'count': len(self.latencies),
            'mean':  statistics.mean(self.latencies),
            'p50':   statistics.median(self.latencies),
            'p95':   sorted(self.latencies)[int(0.95 * len(self.latencies))],
            'p99':   sorted(self.latencies)[int(0.99 * len(self.latencies))],
            'max':   max(self.latencies)
        }


monitor = LatencyMonitor()

# Simulate inference calls with some slow outliers
import random
random.seed(42)
for i in range(200):
    latency = random.gauss(5, 1)        # Normal ~5ms
    if random.random() < 0.02:
        latency = random.gauss(50, 5)    # 2% slow requests
    monitor.record(abs(latency))

stats = monitor.stats()
print("Latency statistics (ms):")
for k, v in stats.items():
    print(f"  {k:8s}: {v:.2f}")

# Alert if p99 exceeds SLA
SLA_P99_MS = 30
if stats['p99'] > SLA_P99_MS:
    print(f"\n[ALERT] p99 latency {stats['p99']:.1f}ms exceeds SLA of {SLA_P99_MS}ms")

Input Drift Detection

Input Feature Drift Detector
Z-score based drift detection. A feature's mean shifting more than 3 standard deviations from training distribution is a strong signal something has changed. Tap on a feature to shift its mean.

python
class DriftMonitor:
    """
    Detect input distribution drift by comparing live feature stats
    to reference (training) stats.
    Uses population stability index (PSI) as the drift metric.
    """

    def __init__(self, reference_mean: torch.Tensor, reference_std: torch.Tensor,
                 n_features: int, z_threshold: float = 3.0):
        self.ref_mean    = reference_mean
        self.ref_std     = reference_std
        self.z_threshold = z_threshold

    def check(self, batch: torch.Tensor) -> dict:
        """Check if batch features have drifted from reference."""
        batch_mean = batch.mean(dim=0)
        batch_std  = batch.std(dim=0)

        # Z-score: how many std deviations is the live mean from reference?
        z_scores = (batch_mean - self.ref_mean).abs() / (self.ref_std + 1e-6)

        drifted_features = (z_scores > self.z_threshold).nonzero(as_tuple=True)[0].tolist()

        return {
            'drift_detected':   len(drifted_features) > 0,
            'drifted_features': drifted_features,
            'max_z_score':      z_scores.max().item(),
            'z_scores':         z_scores
        }


# Simulate drift detection
ref_mean = torch.zeros(10)
ref_std  = torch.ones(10)
drift_m  = DriftMonitor(ref_mean, ref_std, n_features=10)

# Normal batch — no drift
normal_batch = torch.randn(100, 10)
result = drift_m.check(normal_batch)
print(f"Normal batch: drift={result['drift_detected']}  max_z={result['max_z_score']:.2f}")

# Drifted batch — feature 3 and 7 shifted
drifted_batch = torch.randn(100, 10)
drifted_batch[:, 3] += 5.0   # Big shift in feature 3
drifted_batch[:, 7] += 4.0   # Shift in feature 7

result = drift_m.check(drifted_batch)
print(f"Drifted batch: drift={result['drift_detected']}  max_z={result['max_z_score']:.2f}")
print(f"Drifted features: {result['drifted_features']}")

class ModelVersionManager:
    """
    Manages multiple model versions for A/B testing and safe rollback.
    """

    def __init__(self):
        self.versions = {}        # name → model
        self.traffic  = {}        # name → traffic fraction
        self.metrics  = {}        # name → {calls, errors, mean_latency}
        self.active_v = None

    def register(self, name: str, model, traffic_pct: float = 0.0):
        self.versions[name] = model
        self.traffic[name]  = traffic_pct
        self.metrics[name]  = {'calls': 0, 'errors': 0, 'latencies': []}
        print(f"Registered model: {name} ({traffic_pct*100:.0f}% traffic)")

    def route(self) -> str:
        """Route a request based on traffic percentages."""
        r   = random.random()
        cum = 0.0
        for name, pct in self.traffic.items():
            cum += pct
            if r < cum:
                return name
        return self.active_v

    def record_call(self, name: str, latency_ms: float, success: bool):
        m = self.metrics[name]
        m['calls']    += 1
        m['latencies'].append(latency_ms)
        if not success:
            m['errors'] += 1

    def summary(self):
        print("\nModel performance summary:")
        for name, m in self.metrics.items():
            if m['calls'] == 0: continue
            error_rate = m['errors'] / m['calls']
            mean_lat   = statistics.mean(m['latencies']) if m['latencies'] else 0
            print(f"  {name}: calls={m['calls']}, error_rate={error_rate:.3f}, "
                  f"mean_lat={mean_lat:.1f}ms, traffic={self.traffic[name]*100:.0f}%")

    def promote(self, name: str):
        """Promote a model to 100% traffic (after successful A/B test)."""
        for k in self.traffic:
            self.traffic[k] = 0.0
        self.traffic[name] = 1.0
        self.active_v      = name
        print(f"Promoted {name} to 100% traffic.")

    def rollback(self, name: str):
        """Emergency rollback to a previous version."""
        self.promote(name)
        print(f"[ROLLBACK] reverted to {name}")


# Simulate A/B test
vm = ModelVersionManager()
vm.register("v1", None, traffic_pct=0.9)   # 90% to stable v1
vm.register("v2", None, traffic_pct=0.1)   # 10% to new v2

# Simulate traffic
random.seed(0)
for _ in range(1000):
    version = vm.route()
    latency = random.gauss(5, 1) if version == "v1" else random.gauss(4, 0.8)
    success = random.random() > (0.01 if version == "v1" else 0.02)
    vm.record_call(version, abs(latency), success)

vm.summary()

# v2 is better — promote it
vm.promote("v2")
vm.summary()
