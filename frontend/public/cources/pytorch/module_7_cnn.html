SECTION 01

Convolution Intuition & Receptive Field

What is a Convolution?

A convolution slides a small kernel (filter) over an input image. At each position, it computes a dot product between the kernel and the local patch of pixels — producing a feature map that highlights wherever that pattern appears.

Real-World Analogy: A Flashlight Scanning a Document

Imagine shining a small flashlight across a page and noting how bright each patch is relative to a template. A horizontal-edge kernel lights up wherever horizontal edges appear. A vertical-edge kernel lights up where vertical edges appear. Stack enough kernels and you detect textures, shapes, and eventually objects.

Receptive Field

The receptive field is the region of the input that influences a single output neuron. Each conv layer expands the receptive field. Deeper networks see larger context.

def compute_receptive_field(layers):
    """Compute receptive field given a list of (kernel_size, stride) pairs."""
    rf, stride_so_far = 1, 1
    for k, s in layers:
        rf = rf + (k - 1) * stride_so_far
        stride_so_far *= s
    return rf

# 5-layer CNN with 3×3 kernels, stride 1
layers_3x3 = [(3, 1)] * 5
print(f"5-layer 3×3 network RF: {compute_receptive_field(layers_3x3)}×{compute_receptive_field(layers_3x3)}")

# ResNet-style with some stride-2 layers
resnet_layers = [(7, 2), (3, 2), (3, 2), (3, 1), (3, 1)]
print(f"ResNet-style RF: {compute_receptive_field(resnet_layers)}")

class BasicCNN(nn.Module):
    """
    A simple CNN for small image classification.
    Architecture: Conv → BN → ReLU → Pool (repeated) → FC head
    """

    def __init__(self, in_channels: int = 1, n_classes: int = 10):
        super().__init__()

        self.features = nn.Sequential(
            # Block 1
            nn.Conv2d(in_channels, 32, kernel_size=3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.MaxPool2d(2, 2),              # Halve spatial dims

            # Block 2
            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.MaxPool2d(2, 2),

            # Block 3
            nn.Conv2d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((4, 4)),   # Fixed output size regardless of input
        )

        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128 * 4 * 4, 256),
            nn.ReLU(),
            nn.Dropout(0.4),
            nn.Linear(256, n_classes)
        )

    def forward(self, x):
        return self.classifier(self.features(x))


# Test forward pass
cnn    = BasicCNN(in_channels=1, n_classes=10)
x_img  = torch.randn(8, 1, 32, 32)    # Batch=8, grayscale, 32×32
output = cnn(x_img)
print("CNN output shape:", output.shape)   # (8, 10)

# Parameter count
total = sum(p.numel() for p in cnn.parameters())
print(f"Parameters: {total:,}")

SECTION 03

ResNet Intuition

The key insight of ResNet: skip connections let gradients flow directly to earlier layers, solving vanishing gradients in deep networks.

python
# The key insight of ResNet: skip connections let gradients flow directly
# to earlier layers, solving vanishing gradients in deep networks

class ResidualBlock(nn.Module):
    """
    A basic ResNet block: F(x) + x
    The network learns the RESIDUAL (the difference) rather than the full mapping.
    """

    def __init__(self, channels: int):
        super().__init__()
        self.block = nn.Sequential(
            nn.Conv2d(channels, channels, 3, padding=1, bias=False),
            nn.BatchNorm2d(channels),
            nn.ReLU(),
            nn.Conv2d(channels, channels, 3, padding=1, bias=False),
            nn.BatchNorm2d(channels),
        )
        self.relu = nn.ReLU()

    def forward(self, x):
        return self.relu(self.block(x) + x)    # ← Skip connection


# Show that gradients flow through the skip connection even if block saturates
block = ResidualBlock(32)
x     = torch.randn(4, 32, 16, 16, requires_grad=True)
out   = block(x)
loss  = out.sum()
loss.backward()
print("Gradient norm with skip connection:", x.grad.norm().item())
Gradient norm with skip connection: 154.57020568847656
WHY SKIP CONNECTIONS WORK
Without the skip connection, deep networks suffer from vanishing gradients — gradients shrink as they propagate backward through many layers, leaving early layers nearly untrained. The + x in F(x) + x creates a direct path for gradients to flow back to the input, regardless of how poorly the residual block F is trained.
SECTION 04

Transfer Learning: Freezing & Unfreezing

The Core Idea

Training a CNN from scratch on small datasets almost always fails. Instead:

  1. Start with a pretrained model (ImageNet-trained ResNet, EfficientNet, etc.)
  2. Freeze the backbone — don't update the feature extractor weights.
  3. Train only the classification head — fast, data-efficient.
  4. Fine-tune (optional): unfreeze deeper layers and train end-to-end with a small LR.

# Step 3: Fine-tuning — unfreeze progressively

def unfreeze_layers(model, n_layers_from_end: int):
    """Unfreeze the last n convolutional blocks for fine-tuning."""
    all_layers = list(model.children())
    for layer in all_layers[-n_layers_from_end:]:
        for param in layer.parameters():
            param.requires_grad = True

TIP — Transfer Learning Recipe
1. Freeze backbone → train head for 5–10 epochs.
2. Unfreeze last 1–2 blocks → fine-tune at 1/10th the LR.
3. Unfreeze full backbone → fine-tune at 1/100th the LR (optional, only if data is large enough).

def classification_metrics(logits: torch.Tensor, targets: torch.Tensor,
                           n_classes: int) -> dict:
    """Per-class precision, recall, F1 and macro/weighted averages."""
    preds = logits.argmax(dim=1)
    metrics = {}

    per_class_f1 = []
    for c in range(n_classes):
        tp = ((preds == c) & (targets == c)).sum().float()
        fp = ((preds == c) & (targets != c)).sum().float()
        fn = ((preds != c) & (targets == c)).sum().float()
        p  = tp / (tp + fp + 1e-8)
        r  = tp / (tp + fn + 1e-8)
        f1 = 2*p*r / (p + r + 1e-8)
        per_class_f1.append(f1.item())
        metrics[f'class_{c}_f1'] = f1.item()

    metrics['accuracy'] = (preds == targets).float().mean().item()
    metrics['macro_f1'] = np.mean(per_class_f1)

    # Top-k accuracy
    top_k = min(3, n_classes)
    topk_preds = logits.topk(top_k, dim=1).indices
    metrics[f'top{top_k}_acc'] = topk_preds.eq(targets.unsqueeze(1)).any(dim=1).float().mean().item()

    return metrics


# Demo

for k, v in m.items():

Defect / Document Classifier with Transfer Learning

python
import torchvision.models as models
from torch.utils.data import Dataset, DataLoader

# Simulate a document image dataset
# In production: replace with real image paths + PIL loading

class SyntheticImageDataset(Dataset):
    """Synthetic stand-in for a real image dataset."""

    def __init__(self, n_samples: int = 500, n_classes: int = 4,
                 img_size: int = 64):
        self.n_classes = n_classes
        # Simulate features extracted from images (e.g., texture histograms)
        self.X = torch.randn(n_samples, 3, img_size, img_size)
        self.y = torch.randint(0, n_classes, (n_samples,))

    def __len__(self): return len(self.X)

    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]


class TransferLearningClassifier(nn.Module):

    def __init__(self, n_classes: int, backbone_name: str = 'resnet18'):
        super().__init__()
        backbone = models.resnet18(weights=None)

        # Adapt first conv for small images (no pretrained weights here)
        backbone.conv1   = nn.Conv2d(3, 64, kernel_size=3, padding=1, bias=False)
        backbone.maxpool = nn.Identity()   # Skip maxpool for small images

        self.features = nn.Sequential(*list(backbone.children())[:-1])          # Remove FC
        self.head     = nn.Sequential(
            nn.Flatten(),
            nn.Linear(512, 128), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(128, n_classes)
        )

    def forward(self, x):
        return self.head(self.features(x))


N_CLASSES = 4
train_ds = SyntheticImageDataset(400, N_CLASSES, img_size=64)
val_ds   = SyntheticImageDataset(100, N_CLASSES, img_size=64)
train_dl = DataLoader(train_ds, batch_size=32, shuffle=True)
val_dl   = DataLoader(val_ds,   batch_size=64)

model     = TransferLearningClassifier(N_CLASSES)
opt       = torch.optim.AdamW(model.parameters(), lr=1e-3)
criterion = nn.CrossEntropyLoss()
sched     = torch.optim.lr_scheduler.OneCycleLR(
    opt, max_lr=1e-2, steps_per_epoch=len(train_dl), epochs=20)

print(f"Model params: {sum(p.numel() for p in model.parameters()):,}")

Training Loop with OneCycleLR + Gradient Clipping + Best-State Tracking

python
# Training
best_val_acc, best_state = 0.0, None

for epoch in range(20):
    model.train()
    for X_b, y_b in train_dl:
        opt.zero_grad()
        loss = criterion(model(X_b), y_b)
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        opt.step()
        sched.step()

    model.eval()
    with torch.no_grad():
        val_logits = torch.cat([model(X) for X, _ in val_dl])
        val_labels = torch.cat([y         for _, y in val_dl])
        val_acc    = (val_logits.argmax(1) == val_labels).float().mean().item()

    if val_acc > best_val_acc:
        best_val_acc = val_acc
        best_state   = {k: v.clone() for k, v in model.state_dict().items()}

    if epoch % 5 == 0:
        print(f"Epoch {epoch:2d}: val_acc={val_acc:.3f}")

model.load_state_dict(best_state)

# Final evaluation
model.eval()
with torch.no_grad():
    all_logits = torch.cat([model(X) for X, _ in val_dl])
    all_labels = torch.cat([y for _, y in val_dl])

final_metrics = classification_metrics(all_logits, all_labels, N_CLASSES)
print(f"\nFinal model: accuracy={final_metrics['accuracy']:.3f}, "
      f"macro_f1={final_metrics['macro_f1']:.3f}")

# Production inference script
def predict_image_batch(model: nn.Module, images: torch.Tensor,
                        class_names: list) -> list:
    """Run inference on a batch of images."""
    model.eval()
    with torch.no_grad():
        logits = model(images)
        probs  = torch.softmax(logits, dim=1)
        preds  = probs.argmax(dim=1)

    results = []
    for i in range(len(images)):
        results.append({
            'class':      class_names[preds[i].item()],
            'confidence': probs[i, preds[i]].item(),
            'all_probs':  {class_names[j]: probs[i, j].item()
                            for j in range(len(class_names))}
        })
    return results


class_names = ['scratch', 'dent', 'crack', 'ok']
test_imgs   = torch.randn(4, 3, 64, 64)
results     = predict_image_batch(model, test_imgs, class_names)

print("\nInference results:")
for i, r in enumerate(results):
    print(f"  Image {i}: {r['class']} (conf={r['confidence']:.3f})")
