SECTION 01

Why Sequence Modelling Matters in DS / AIOps

The Problem with Standard MLPs for Sequences

An MLP treats each input independently. Feed it [login, file_read, file_write] and [file_write, login, file_read] — it doesn't know these have different temporal meaning.

In AIOps:

  • A permission_denied after sudo_attempt is very different from permission_denied after file_read.
  • The order and context of events matters enormously for anomaly detection.

SECTION 02

LSTM & GRU: When They Still Win

The LSTM Cell Intuition

An LSTM maintains two states: cell state c_t — long-term memory (the "conveyor belt"); hidden state h_t — short-term working memory. Three gates control what to keep, forget, and write.

TIP — When LSTM/GRU Beat Transformers
Sequences are very long (> 10k tokens) and attention is prohibitively expensive. Online/streaming inference — process one token at a time with constant memory. Tiny datasets — Transformers need more data to generalise. Deployment constraints — LSTMs are simpler to export and run on edge devices.
SECTION 03

Attention Intuition

The Core Idea

Instead of compressing the entire sequence into a fixed-size hidden state, attention lets the model look back at any position in the sequence when computing each output.

Real-World Analogy: Reading a log to detect an anomaly. Rather than summarising all 1000 events into a single vector, attention lets the model say: "To understand event 850, let me focus on events 10, 200, and 847 — they're the most relevant context."

# Scaled dot-product attention — the heart of the Transformer
def scaled_dot_product_attention(Q, K, V, mask=None):
    """
    Q: (B, heads, T, d_k) — queries
    K: (B, heads, T, d_k) — keys
    V: (B, heads, T, d_v) — values
    Returns: attended values + attention weights
    """
    d_k    = Q.shape[-1]
    scores = torch.matmul(Q, K.transpose(-2, -1)) / (d_k ** 0.5)   # Scale

    if mask is not None:
        scores = scores.masked_fill(mask == 0, float('-inf'))

    weights = torch.softmax(scores, dim=-1)   # (B, heads, T, T)
    output  = torch.matmul(weights, V)         # (B, heads, T, d_v)
    return output, weights


# Demo: 4 heads, seq_len=5, d_k=8
B, heads, T, d_k = 2, 4, 5, 8
Q = torch.randn(B, heads, T, d_k)
K = torch.randn(B, heads, T, d_k)
V = torch.randn(B, heads, T, d_k)

out, weights = scaled_dot_product_attention(Q, K, V)
print("Attention output shape: ", out.shape)     # (2, 4, 5, 8)
print("Attention weights shape:", weights.shape) # (2, 4, 5, 5) — each token attends to all
print("Attention weights sum (should be 1.0):", weights[0, 0, 0].sum().item())

class PositionalEncoding(nn.Module):
    """
    Sinusoidal positional encoding.
    Adds position information to embeddings so the model knows token order.
    """

    def __init__(self, d_model: int, max_seq_len: int = 512, dropout: float = 0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)

        # Build the positional encoding matrix
        pe  = torch.zeros(max_seq_len, d_model)
        pos = torch.arange(0, max_seq_len).unsqueeze(1).float()
        div = torch.exp(torch.arange(0, d_model, 2).float() *
                        -(np.log(10000.0) / d_model))

        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        pe = pe.unsqueeze(0)              # (1, max_seq_len, d_model)
        self.register_buffer('pe', pe)   # Non-trainable buffer

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (B, T, d_model)
        x = x + self.pe[:, :x.size(1), :]
        return self.dropout(x)


class LogTransformer(nn.Module):
    """
    Transformer encoder for log event sequences.
    Input: sequence of log event IDs
    Output: per-token representations OR sequence-level representation
    """

    def __init__(self, vocab_size: int, d_model: int = 128, n_heads: int = 4,
                 n_layers: int = 3, d_ff: int = 256, dropout: float = 0.1,
                 max_seq_len: int = 64):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model, padding_idx=0)
        self.pos_enc   = PositionalEncoding(d_model, max_seq_len, dropout)

        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=n_heads, dim_feedforward=d_ff,
            dropout=dropout, batch_first=True, norm_first=True   # Pre-norm: more stable
        )
        self.encoder = nn.TransformerEncoder(encoder_layer, num_layers=n_layers)
        self.norm    = nn.LayerNorm(d_model)

    def forward(self, token_ids: torch.Tensor,
                padding_mask: torch.Tensor = None) -> torch.Tensor:
        """
        token_ids:    (B, T) — integer event IDs
        padding_mask: (B, T) — True where tokens are padding
        Returns:      (B, T, d_model) — contextual representations
        """
        x = self.pos_enc(self.embedding(token_ids))
        x = self.encoder(x, src_key_padding_mask=padding_mask)
        return self.norm(x)

Padding Mask

from torch.nn.utils.rnn import pad_sequence

def make_padding_mask(sequences: list, pad_id: int = 0) -> torch.Tensor:
    """Create a padding mask: True where tokens are padding."""
    padded = pad_sequence(sequences, batch_first=True, padding_value=pad_id)
    mask   = (padded == pad_id)
    return padded, mask


seqs = [
    torch.tensor([3, 7, 12, 5]),

Causal Mask for Autoregressive Generation

python
def make_causal_mask(seq_len: int) -> torch.Tensor:
    """
    Upper-triangular mask for autoregressive (causal) attention.
    Position i can only attend to positions 0..i (not future tokens).
    """
    # In PyTorch TransformerDecoder: True means 'ignore this position'
    mask = torch.triu(torch.ones(seq_len, seq_len), diagonal=1).bool()
    return mask

mask = make_causal_mask(6)
print("Causal mask (True = masked):")
print(mask.int())

HANDS-ON PROJECT

Log-Sequence Prediction + Anomaly Scoring

Two tasks in one model:

  1. Next event prediction — predict the next log event ID given previous events.
  2. Anomaly scoring — high loss on a sequence = anomalous sequence.
python
# Dataset
torch.manual_seed(77)
np.random.seed(77)

def generate_log_sequences(n_seqs=2000, vocab_size=50, seq_len=20,
                            anomaly_rate=0.1):
    """
    Normal sequences: Markov chain over events 1-30.
    Anomalous sequences: random events from full vocab (including rare 31-49).
    """
    sequences, labels = [], []
    trans = np.random.dirichlet(np.ones(30) * 0.5, size=30)   # Transition matrix

    for _ in range(n_seqs):
        is_anomaly = np.random.rand() < anomaly_rate
        seq = []
        state = np.random.randint(0, 30)
        for _ in range(seq_len):
            if is_anomaly:
                event = np.random.randint(1, vocab_size)
            else:
                event = int(np.random.choice(30, p=trans[state])) + 1
            seq.append(event)
            state = event - 1 if not is_anomaly else np.random.randint(0, 30)
        sequences.append(torch.tensor(seq, dtype=torch.long))
        labels.append(int(is_anomaly))

    return sequences, torch.tensor(labels)


seqs, labels = generate_log_sequences(2000, vocab_size=50, seq_len=20)
print(f"Sequences: {len(seqs)}, Anomaly rate: {labels.float().mean():.3f}")

Next-Event Dataset and Predictor

from torch.utils.data import Dataset, DataLoader

class NextEventDataset(Dataset):
    """Each sample: input = seq[:-1], target = seq[1:] (next event prediction)."""

    def __init__(self, sequences: list, labels: torch.Tensor):
        self.sequences = sequences
        self.labels    = labels

    def __len__(self): return len(self.sequences)

    def __getitem__(self, idx):
        seq = self.sequences[idx]
        return seq[:-1], seq[1:], self.labels[idx]


dataset = NextEventDataset(seqs, labels)

split    = int(0.8 * len(dataset))
train_ds = torch.utils.data.Subset(dataset, range(split))
val_ds   = torch.utils.data.Subset(dataset, range(split, len(dataset)))

train_dl = DataLoader(train_ds, batch_size=64, shuffle=True)
val_dl   = DataLoader(val_ds,   batch_size=128)

# Next-event prediction model
class NextEventPredictor(nn.Module):

    def __init__(self, vocab_size=50, d_model=64, n_heads=4, n_layers=2):
        super().__init__()
        self.transformer = LogTransformer(
            vocab_size=vocab_size + 1,   # +1 for padding
            d_model=d_model, n_heads=n_heads, n_layers=n_layers,
            d_ff=128, max_seq_len=32
        )
        self.head = nn.Linear(d_model, vocab_size + 1)

    def forward(self, x):
        # Causal mask: each position only attends to previous positions
        T    = x.size(1)
        mask = torch.triu(torch.ones(T, T, device=x.device), diagonal=1).bool()
        h    = self.transformer.embedding(x)
        h    = self.transformer.pos_enc(h)
        h    = self.transformer.encoder(h, mask=mask)
        h    = self.transformer.norm(h)
        return self.head(h)   # (B, T, vocab_size)


pred_model = NextEventPredictor(vocab_size=50)
opt        = torch.optim.AdamW(pred_model.parameters(), lr=1e-3)
criterion  = nn.CrossEntropyLoss(ignore_index=0)
sched      = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=30)

Training Loop with Reshape for Per-Token Loss

python
# Training
for epoch in range(30):
    pred_model.train()
    total_loss = 0
    for x_b, y_b, _ in train_dl:
        opt.zero_grad()
        logits = pred_model(x_b)              # (B, T, vocab)
        # Reshape for CrossEntropyLoss: (B*T, vocab), (B*T,)
        loss = criterion(logits.reshape(-1, logits.size(-1)), y_b.reshape(-1))
        loss.backward()
        torch.nn.utils.clip_grad_norm_(pred_model.parameters(), 1.0)
        opt.step()
        total_loss += loss.item()
    sched.step()

    if epoch % 10 == 0:
        print(f"Epoch {epoch}: loss={total_loss/len(train_dl):.4f}")

# Anomaly scoring via reconstruction loss
def anomaly_score(model: nn.Module, sequences: torch.Tensor) -> torch.Tensor:
    """
    Compute per-sequence anomaly score = mean cross-entropy loss.
    High loss → sequence is unusual relative to training distribution → anomaly.
    """
    model.eval()
    with torch.no_grad():
        x = sequences[:, :-1]
        y = sequences[:, 1:]
        logits = model(x)
        # Per-token loss
        losses = nn.CrossEntropyLoss(reduction='none')(
            logits.reshape(-1, logits.size(-1)),
            y.reshape(-1)
        ).reshape(sequences.size(0), -1)
        return losses.mean(dim=1)   # Per-sequence mean loss


# Score validation sequences
val_seqs   = torch.stack([seqs[i] for i in range(split, len(seqs))])
val_labels = labels[split:]

scores = anomaly_score(pred_model, val_seqs)
print(f"Normal seq mean score:    {scores[val_labels==0].mean():.4f}")
print(f"Anomalous seq mean score: {scores[val_labels==1].mean():.4f}")

# Choose threshold on val set (e.g., 95th percentile of normal scores)
normal_scores = scores[val_labels == 0]
threshold     = normal_scores.quantile(0.95).item()
predictions   = (scores > threshold).long()
precision_a   = ((predictions == 1) & (val_labels == 1)).sum().float() / (predictions == 1).sum().clamp(1)
recall_a      = ((predictions == 1) & (val_labels == 1)).sum().float() / (val_labels == 1).sum().clamp(1)
print(f"\nAnomaly detection @ 95th-pct threshold ({threshold:.3f}):")
print(f"  Precision: {precision_a:.3f}")
print(f"  Recall:    {recall_a:.3f}")
