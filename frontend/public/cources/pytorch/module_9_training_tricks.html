import torch
import torch.nn as nn
import numpy as np
torch.manual_seed(42)

# Dropout randomly zeros activations during training
# At inference, all neurons are active but outputs are scaled by (1-p)

# Where to put dropout:
# - After activation functions in hidden layers
# - Before the final linear layer
# - NOT after the very last linear layer (before loss)

class RegularisedMLP(nn.Module):

    def __init__(self, in_dim, hidden, out_dim, dropout_p=0.3):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, hidden), nn.ReLU(), nn.Dropout(dropout_p),
            nn.Linear(hidden, hidden), nn.ReLU(), nn.Dropout(dropout_p),
            nn.Linear(hidden, out_dim)
        )

    def forward(self, x): return self.net(x)

Weight Decay: L2 Regularisation

python
# Weight decay penalises large weights — simpler models generalise better
# AdamW implements DECOUPLED weight decay (correct version)

model = RegularisedMLP(10, 64, 3)

# Apply different weight decay to different parameter groups
# Common pattern: no weight decay on biases and LayerNorm parameters
no_decay     = ['bias', 'norm']
decay_params = [p for n, p in model.named_parameters()
                if not any(nd in n for nd in no_decay)]
nodecay_params = [p for n, p in model.named_parameters()
                  if any(nd in n for nd in no_decay)]

opt = torch.optim.AdamW([
    {'params': decay_params,    'weight_decay': 1e-2},
    {'params': nodecay_params,  'weight_decay': 0.0},
], lr=1e-3)

print("Optimizer with selective weight decay configured.")
print(f"Decay param count:    {sum(p.numel() for p in decay_params):,}")
print(f"No-decay param count: {sum(p.numel() for p in nodecay_params):,}")

Augmentations: Mixup for Tabular / Time-Series

Mixup creates new training samples by linearly interpolating between two real samples. Originally proposed for images, it works well for tabular and time-series data too.

def mixup_batch(X: torch.Tensor, y: torch.Tensor, alpha: float = 0.4):
    """
    Apply mixup augmentation to a batch.
    Creates convex combinations of pairs of samples.

    Returns:
        X_mixed: interpolated features
        y_a, y_b: original labels
        lam:      mixing coefficient
    """
    lam   = np.random.beta(alpha, alpha) if alpha > 0 else 1.0
    idx   = torch.randperm(len(X))

    X_mixed = lam * X + (1 - lam) * X[idx]
    y_a     = y
    y_b     = y[idx]
    return X_mixed, y_a, y_b, lam


def mixup_criterion(criterion, pred, y_a, y_b, lam):
    """Loss for mixup-augmented samples."""
    return lam * criterion(pred, y_a) + (1 - lam) * criterion(pred, y_b)


# Usage in training loop
X_b = torch.randn(32, 10)
y_b = torch.randint(0, 3, (32,))
criterion_ce = nn.CrossEntropyLoss()
model_m = RegularisedMLP(10, 64, 3)

X_mix, y_a, y_b_mix, lam = mixup_batch(X_b, y_b, alpha=0.4)
logits = model_m(X_mix)
loss   = mixup_criterion(criterion_ce, logits, y_a, y_b_mix, lam)
print(f"Mixup loss: {loss.item():.4f}  (lambda={lam:.3f})")
Mixup loss: 1.1269 (lambda=0.001)
SECTION 04

Class Imbalance: Focal Loss + Weighted Sampling

Focal Loss

Focal loss down-weights easy, well-classified examples so the model focuses on hard ones. Originally designed for object detection, it's highly useful for anomaly detection and fraud.

Focal Loss vs Cross-Entropy
FL(p_t) = -α (1 − p_t)^γ log(p_t). The cross-entropy curve treats all examples the same way; focal loss with γ > 0 down-weights easy ones (high p_t) so training focuses on hard examples.

class FocalLoss(nn.Module):
    """
    Focal Loss: FL(p_t) = -alpha_t * (1 - p_t)^gamma * log(p_t)
    gamma > 0 reduces loss for well-classified examples.
    alpha balances class importance.
    """

    def __init__(self, alpha: float = 0.25, gamma: float = 2.0,
                 reduction: str = 'mean'):
        super().__init__()
        self.alpha     = alpha
        self.gamma     = gamma
        self.reduction = reduction

    def forward(self, logits: torch.Tensor, targets: torch.Tensor) -> torch.Tensor:
        ce_loss = nn.functional.cross_entropy(logits, targets, reduction='none')
        pt      = torch.exp(-ce_loss)             # p_t = probability of correct class
        focal   = self.alpha * (1 - pt) ** self.gamma * ce_loss

        if self.reduction == 'mean': return focal.mean()
        if self.reduction == 'sum':  return focal.sum()
        return focal


# Compare on imbalanced batch
logits  = torch.randn(32, 2)
targets = torch.zeros(32, dtype=torch.long)
targets[:3] = 1   # Only 3 positives out of 32 — heavily imbalanced

ce_loss    = nn.CrossEntropyLoss()(logits, targets)
focal_loss = FocalLoss(gamma=2.0)(logits, targets)

print(f"Cross-entropy loss: {ce_loss.item():.4f}")
print(f"Focal loss:         {focal_loss.item():.4f}")

SECTION 05

Self-Supervised Learning: Contrastive Learning Intro

Contrastive learning trains a model to bring similar samples together and push different samples apart — without requiring labels for every sample.

python
class ProjectionHead(nn.Module):
    """Small MLP head used for contrastive learning."""
    def __init__(self, in_dim, proj_dim=64):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, in_dim), nn.ReLU(),
            nn.Linear(in_dim, proj_dim)
        )
    def forward(self, x): return self.net(x)


def nt_xent_loss(z1: torch.Tensor, z2: torch.Tensor, temperature: float = 0.5):
    """
    NT-Xent (Normalised Temperature-scaled Cross Entropy) loss.
    z1, z2: two augmented views of the same samples (B, D).
    Positives: (z1[i], z2[i]). Negatives: all other pairs.
    """
    B = z1.size(0)
    z = torch.cat([z1, z2], dim=0)                       # (2B, D)
    z = nn.functional.normalize(z, dim=1)

    sim = torch.matmul(z, z.T) / temperature              # (2B, 2B)
    # Remove self-similarity from denominator
    mask = torch.eye(2*B, dtype=torch.bool, device=z.device)
    sim.masked_fill_(mask, float('-inf'))

    # Positive pairs: i with i+B and i+B with i
    labels = torch.cat([torch.arange(B) + B,
                        torch.arange(B)], dim=0).to(z.device)

    return nn.CrossEntropyLoss()(sim, labels)


# Demo: augment log feature vectors by adding noise
B, D = 32, 64
z1 = torch.randn(B, D)
z2 = z1 + torch.randn(B, D) * 0.1   # Slightly perturbed (positive pair)

proj_head = ProjectionHead(D, proj_dim=32)
z1_proj   = proj_head(z1)
z2_proj   = proj_head(z2)

loss = nt_xent_loss(z1_proj, z2_proj, temperature=0.5)
print(f"NT-Xent contrastive loss: {loss.item():.4f}")
NT-Xent contrastive loss: 2.7913
WHY CONTRASTIVE PRETRAINING IS POWERFUL
Most production datasets have far more unlabelled samples than labelled ones. Contrastive learning extracts general-purpose features from unlabelled data using simple augmentations (noise, masking, time-shift). You then attach a small classifier head and fine-tune on the small labelled set — often beating supervised-only baselines.

# Optuna is the recommended HPO library for PyTorch
# pip install optuna

# Without Optuna installed, here is the design pattern:

def objective(trial_config: dict) -> float:
    """
    One trial: build + train model with given hyperparameters.
    Returns: validation metric (to minimise or maximise).
    """
    lr      = trial_config['lr']
    dropout = trial_config['dropout']
    hidden  = trial_config['hidden_dim']
    wd      = trial_config['weight_decay']

    X_tr = torch.randn(500, 10); y_tr = torch.randint(0, 3, (500,))
    X_vl = torch.randn(100, 10); y_vl = torch.randint(0, 3, (100,))

    model = RegularisedMLP(10, hidden, 3, dropout_p=dropout)
    opt   = torch.optim.AdamW(model.parameters(), lr=lr, weight_decay=wd)
    crit  = nn.CrossEntropyLoss(label_smoothing=0.1)

    for _ in range(20):
        model.train(); opt.zero_grad()
        loss = crit(model(X_tr), y_tr)
        loss.backward(); opt.step()

    model.eval()
    with torch.no_grad():
        acc = (model(X_vl).argmax(1) == y_vl).float().mean().item()

    return acc


# Manual grid search (replace with Optuna for production)
search_space = [
    {'lr': 1e-3, 'dropout': 0.2, 'hidden_dim': 64,  'weight_decay': 1e-3},
    {'lr': 5e-4, 'dropout': 0.3, 'hidden_dim': 128, 'weight_decay': 1e-2},
    {'lr': 1e-3, 'dropout': 0.1, 'hidden_dim': 32,  'weight_decay': 1e-4},
]

print("Manual HPO search:")
for i, config in enumerate(search_space):
    val_acc = objective(config)
    print(f"  Trial {i}: lr={config['lr']}, hidden={config['hidden_dim']} "
          f"→ val_acc={val_acc:.3f}")

Module Summary

ConceptKey Takeaway
DropoutAdd after activations in hidden layers. Never after the final layer.
Label smoothingCrossEntropyLoss(label_smoothing=0.1). Prevents overconfidence.
