SECTION 01

nn.Module, Parameters, and Buffers

What is nn.Module?

nn.Module is the base class for every neural network in PyTorch. It gives your model four superpowers:

  1. Parameter tracking — any nn.Parameter or sub-module you assign is automatically registered and returned by .parameters().
  2. State saving/loading — .state_dict() and .load_state_dict() let you checkpoint and resume.
  3. Device management — .to(device) moves all parameters and buffers at once.
  4. Training/eval modes — .train() and .eval() toggle behaviour of dropout, batchnorm, etc.

Real-World Analogy: A Blueprint vs a Building

A class that inherits from nn.Module is like a blueprint. When you instantiate it, you get an actual building (a model object) with real rooms (parameters) and fixtures (buffers). The blueprint can be used to construct as many independent buildings as you want.

python
import torch
import torch.nn as nn

# The minimal nn.Module
class TinyLinear(nn.Module):

    def __init__(self, in_features: int, out_features: int):
        super().__init__()                                         # Always call this first
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias   = nn.Parameter(torch.zeros(out_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return x @ self.weight.T + self.bias


model = TinyLinear(3, 2)
x     = torch.randn(5, 3)              # Batch of 5 samples
out   = model(x)                       # Calls forward() via __call__

print("Output shape:", out.shape)
print("\nParameters:")
for name, param in model.named_parameters():
    print(f" {name}: shape={param.shape}, requires_grad={param.requires_grad}")
Output shape: torch.Size([5, 2]) Parameters: weight: shape=torch.Size([2, 3]), requires_grad=True bias: shape=torch.Size([2]), requires_grad=True

python
class ModelWithBuffer(nn.Module):

    def __init__(self):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(4, 4))

        # Buffer: moves with .to(device), saved in state_dict, but not trained
        self.register_buffer('running_mean', torch.zeros(4))

        # Plain attribute: NOT tracked by PyTorch
        self.some_constant = 42          # Won't appear in state_dict

    def forward(self, x):
        return x @ self.weight + self.running_mean


m = ModelWithBuffer()
print("Parameters:", [n for n, _ in m.named_parameters()])
print("Buffers:   ", [n for n, _ in m.named_buffers()])
print("State dict keys:", list(m.state_dict().keys()))
Parameters: ['weight'] Buffers: ['running_mean'] State dict keys: ['weight', 'running_mean']
NOTE — When to use a buffer
BatchNorm's running_mean and running_var are classic examples — they track statistics during training but aren't trained via backprop.

SECTION 02

Forward Pass Patterns and Good Style

The Three Patterns

python
# Pattern 1: Sequential (simple, clean for linear stacks)
model_seq = nn.Sequential(
    nn.Linear(10, 64),
    nn.ReLU(),
    nn.Linear(64, 32),
    nn.ReLU(),
    nn.Linear(32, 1)
)

# Pattern 2: Custom class (flexible, recommended for anything non-trivial)
class MLP(nn.Module):

    def __init__(self, in_dim, hidden_dim, out_dim):
        super().__init__()
        self.fc1 = nn.Linear(in_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, out_dim)
        self.act = nn.ReLU()

    def forward(self, x):
        x = self.act(self.fc1(x))
        x = self.fc2(x)
        return x


# Pattern 3: ModuleList for dynamic depth
class DeepMLP(nn.Module):

    def __init__(self, dims: list):
        super().__init__()
        # MUST use nn.ModuleList — plain Python list won't register parameters!
        self.layers = nn.ModuleList([
            nn.Linear(dims[i], dims[i+1]) for i in range(len(dims)-1)
        ])
        self.act = nn.ReLU()

    def forward(self, x):
        for i, layer in enumerate(self.layers):
            x = layer(x)
            if i < len(self.layers) - 1:    # No activation after last layer
                x = self.act(x)
        return x


deep = DeepMLP([10, 64, 32, 16, 1])
x    = torch.randn(8, 10)
print("DeepMLP output:", deep(x).shape)

# Count parameters
total = sum(p.numel() for p in deep.parameters())
train = sum(p.numel() for p in deep.parameters() if p.requires_grad)
print(f"Total params: {total:,} | Trainable: {train:,}")

SECTION 04

Weight Initialisation

Why It Matters

Start with weights that are too large → activations explode → NaN loss.
Start with weights too small → gradients vanish → network doesn't learn.

PyTorch's default initialisations are already good, but understanding them helps you fix broken models.

# Default PyTorch init — Kaiming uniform for Linear layers
layer = nn.Linear(64, 32)
print("Default weight stats:")
print(f"  mean={layer.weight.data.mean():.4f}, std={layer.weight.data.std():.4f}")

# Manual init using torch.nn.init
def init_weights(module):
    if isinstance(module, nn.Linear):
        nn.init.xavier_uniform_(module.weight)     # Good for tanh/sigmoid
        nn.init.zeros_(module.bias)
    elif isinstance(module, nn.Conv2d):
        nn.init.kaiming_normal_(module.weight, mode='fan_out', nonlinearity='relu')


class Net(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(32, 64)
        self.fc2 = nn.Linear(64, 10)
        self.apply(init_weights)                    # Apply to all submodules recursively

net = Net()
print("\nAfter Xavier init:")
print(f"  mean={net.fc1.weight.data.mean():.4f}, std={net.fc1.weight.data.std():.4f}")

SECTION 05

Saving and Loading state_dict Properly

The Correct Pattern

python
import os

model = MLP(10, 64, 3)

# Save ONLY the state dict (not the whole model object)
torch.save(model.state_dict(), "mlp_checkpoint.pt")
print("Saved state_dict keys:", list(model.state_dict().keys()))

# Load: instantiate the architecture, then load weights
model_loaded = MLP(10, 64, 3)                           # Same architecture
state = torch.load("mlp_checkpoint.pt", map_location="cpu")
model_loaded.load_state_dict(state)
model_loaded.eval()                                     # Always set eval for inference

print("Model loaded successfully.")
print("Is in eval mode?", not model_loaded.training)
Saved state_dict keys: ['fc1.weight', 'fc1.bias', 'fc2.weight', 'fc2.bias'] Model loaded successfully. Is in eval mode? True
IMPORTANT
Always save the state_dict, not the whole model. Saving the whole model with torch.save(model, ...) serialises the class definition too, which causes problems if you refactor your code later. The state_dict is just a dictionary of tensors — portable and stable.

Saving Training Checkpoints (Full State)

python

Model Definition

python
class CustomerMLP(nn.Module):

    def __init__(self, in_dim: int, hidden_dims: list, out_dim: int, dropout: float = 0.3):
        super().__init__()

        dims   = [in_dim] + hidden_dims
        layers = []
        for i in range(len(dims) - 1):
            layers.extend([
                nn.Linear(dims[i], dims[i+1]),
                nn.BatchNorm1d(dims[i+1]),
                nn.ReLU(),
                nn.Dropout(dropout)
            ])
        layers.append(nn.Linear(dims[-1], out_dim))

        self.net = nn.Sequential(*layers)
        self.apply(self._init_weights)

    def _init_weights(self, m):
        if isinstance(m, nn.Linear):
            nn.init.kaiming_normal_(m.weight, nonlinearity='relu')
            nn.init.zeros_(m.bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)


model = CustomerMLP(in_dim=12, hidden_dims=[64, 32], out_dim=3)
total_params = sum(p.numel() for p in model.parameters())
print(f"Model: {model}")
print(f"Total parameters: {total_params:,}")

Training

python
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-4)
criterion = nn.CrossEntropyLoss()

best_val_acc = 0.0
best_state   = None

for epoch in range(100):
    # --- Train ---
    model.train()
    optimizer.zero_grad()
    logits = model(X_train)
    loss   = criterion(logits, y_train)
    loss.backward()
    optimizer.step()

    # --- Validate ---
    model.eval()
    with torch.no_grad():
        val_logits = model(X_val)
        val_preds  = val_logits.argmax(dim=1)
        val_acc    = (val_preds == y_val).float().mean().item()

    if val_acc > best_val_acc:
        best_val_acc = val_acc
        best_state   = model.state_dict().copy()           # Save best weights

    if epoch % 20 == 0:
        print(f"Epoch {epoch:3d}: loss={loss.item():.4f}  val_acc={val_acc:.3f}")

print(f"\nBest val accuracy: {best_val_acc:.3f}")

Inference Script (simulate loading in a new session)

def load_model_for_inference(checkpoint_path: str, device: str = "cpu") -> nn.Module:
    """Load a saved CustomerMLP for inference."""
    checkpoint = torch.load(checkpoint_path, map_location=device)
    cfg        = checkpoint['model_config']
    model      = CustomerMLP(**cfg)
    model.load_state_dict(checkpoint['model_state_dict'])
    model.eval()
    return model


def predict(model: nn.Module, X: torch.Tensor) -> dict:
    """Run inference and return predictions + probabilities."""
    with torch.no_grad():
        logits = model(X)
        probs  = torch.softmax(logits, dim=1)
        preds  = probs.argmax(dim=1)
    return {
        'predictions':   preds,
        'probabilities': probs,
        'class_names':   ['low_risk', 'medium_risk', 'high_risk']
    }


# Run inference
loaded_model = load_model_for_inference("customer_mlp_best.pt")
results      = predict(loaded_model, X_val[:5])

print("\nInference results for 5 validation samples:")
for i in range(5):
    pred  = results['predictions'][i].item()
    probs = results['probabilities'][i]
    name  = results['class_names'][pred]
    print(f"  Sample {i}: {name:12s} probs={probs.round(decimals=3)}")

RECAP

Module Summary

ConceptKey Takeaway
nn.ModuleBase class for all models. Always call super().__init__() first.
nn.ParameterTensors that are automatically tracked and trained.
register_bufferNon-trainable state that moves with the model (e.g., running stats).
nn.ModuleListUse instead of plain Python lists to hold submodules.
state_dictSave this, not the whole model. Load into a fresh instance.
model.eval()Always call before inference — disables dropout, fixes batchnorm.
Weight initKaiming for ReLU networks; Xavier for Tanh/Sigmoid.
