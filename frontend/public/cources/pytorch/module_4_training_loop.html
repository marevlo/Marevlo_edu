SECTION 01

The Canonical Training Loop

The Four Steps — Every Time, In Order

This never changes. Every training loop you write, ever, follows this structure.

The Five-Step Loop Walkthrough

Click "Play full loop" or "Step forward" to walk through one iteration.

Real-World Analogy: A Student Sitting Exams

  1. Zero grad — forget last week's exam feedback.
  2. Forward pass — sit the exam (make predictions).
  3. Loss — get your score (how wrong were you?).
  4. Backward pass — analyse which answers were wrong and why.
  5. Optimizer step — adjust your study approach based on feedback. Repeat until the exam score is good.
python
import torch
import torch.nn as nn
import torch.nn.functional as F

torch.manual_seed(42)

# --- Minimal working training loop template ---
def train_one_epoch(model, loader, optimizer, criterion, device):
    model.train()
    total_loss = 0.0

    for X_batch, y_batch in loader:
        X_batch = X_batch.to(device)
        y_batch = y_batch.to(device)

        optimizer.zero_grad()                       # Step 1: clear gradients
        output = model(X_batch)                     # Step 2: forward pass
        loss   = criterion(output, y_batch)         # Step 3: compute loss
        loss.backward()                              # Step 4: backward pass
        optimizer.step()                             # Step 5: update weights

        total_loss += loss.item() * len(X_batch)

    return total_loss / len(loader.dataset)


def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct = 0.0, 0

    with torch.no_grad():
        for X_batch, y_batch in loader:
            X_batch = X_batch.to(device)
            y_batch = y_batch.to(device)
            output = model(X_batch)
            loss   = criterion(output, y_batch)
            total_loss += loss.item() * len(X_batch)
            correct    += (output.argmax(1) == y_batch).sum().item()

    n = len(loader.dataset)
    return total_loss / n, correct / n

print("Training loop template defined.")

Loss Functions

CrossEntropyLoss — Multi-Class Classification

python
# CrossEntropyLoss = LogSoftmax + NLLLoss in one step
# Input: logits of shape (N, C) — raw scores, NOT probabilities
# Target: class indices of shape (N,) — integers in [0, C-1]

criterion = nn.CrossEntropyLoss()

logits = torch.tensor([[2.0, 1.0, 0.5],      # Sample 0 → likely class 0
                       [0.1, 3.0, 0.2],      # Sample 1 → likely class 1
                       [0.2, 0.1, 4.0]])     # Sample 2 → likely class 2
targets = torch.tensor([0, 1, 2])

loss = criterion(logits, targets)
print("CrossEntropy loss:", loss.item())     # Should be low — correct predictions

# Wrong predictions → high loss
targets_wrong = torch.tensor([2, 0, 1])
loss_wrong = criterion(logits, targets_wrong)
print("Wrong prediction loss:", loss_wrong.item())

model_adamw = nn.Linear(10, 1)

# In AdamW, weight decay is applied directly to weights → correct L2 regularisation
opt_adamw = torch.optim.AdamW(
    model_adamw.parameters(),
    lr=1e-3,
    weight_decay=1e-2        # Decoupled weight decay — typically 0.01
)

Learning Rate Scheduling

Why Schedule the LR?

A high LR early on helps explore the loss landscape quickly. A low LR later helps converge to a good minimum without overshooting. LR scheduling automates this.

The Correct Scheduling Pattern

python
# CRITICAL: scheduler.step() goes AFTER optimizer.step()

model     = nn.Linear(10, 1)
opt       = torch.optim.AdamW(model.parameters(), lr=1e-3)
scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=50)

for epoch in range(5):
    # ... training ...
    opt.step()             # ← optimizer first
    scheduler.step()       # ← scheduler second
    print(f"Epoch {epoch}: LR = {opt.param_groups[0]['lr']:.6f}")

python
model = nn.Sequential(nn.Linear(10, 64), nn.ReLU(), nn.Linear(64, 1))
opt   = torch.optim.AdamW(model.parameters())
X     = torch.randn(32, 10)
y     = torch.randn(32, 1)

# Pass 1: measure norm before clipping
opt.zero_grad()
loss = nn.MSELoss()(model(X), y)
loss.backward()
total_norm_before = torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=float('inf'))
print(f"Gradient norm before clipping: {total_norm_before:.4f}")

# Pass 2: fresh forward+backward, then clip to max_norm=1.0
opt.zero_grad()
loss = nn.MSELoss()(model(X), y)   # Recompute — can't backward through freed graph
loss.backward()
total_norm_after = torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
print(f"Gradient norm after clipping: {min(total_norm_after.item(), 1.0):.4f}")
opt.step()

Gradient Accumulation — Simulate Larger Batches

python
# When GPU memory is limited, accumulate gradients over N mini-batches
# before stepping — this simulates a batch size of N × mini_batch_size

model     = nn.Sequential(nn.Linear(10, 32), nn.ReLU(), nn.Linear(32, 1))
opt       = torch.optim.AdamW(model.parameters())
criterion = nn.MSELoss()

accumulate_steps = 4   # Effective batch = 4 × 8 = 32

for step, (X_b, y_b) in enumerate([(torch.randn(8,10), torch.randn(8,1))]*8):
    loss = criterion(model(X_b), y_b)
    loss = loss / accumulate_steps           # Scale loss so gradients are averaged
    loss.backward()

    if (step + 1) % accumulate_steps == 0:
        opt.step()
        opt.zero_grad()
        print(f"Optimizer step at step {step+1}")

Mixed Precision Training

python
# Mixed precision: compute forward/backward in float16, store weights in float32
# Result: ~2x speedup, ~50% memory reduction on compatible GPUs

from torch.amp import autocast, GradScaler

device = "cuda" if torch.cuda.is_available() else "cpu"
model = nn.Sequential(nn.Linear(10, 64), nn.ReLU(), nn.Linear(64, 1)).to(device)
opt    = torch.optim.AdamW(model.parameters())

X = torch.randn(32, 10, device=device)
y = torch.randn(32, 1, device=device)

opt.zero_grad()

if device == "cuda":
    scaler = GradScaler("cuda")
    with autocast(device_type="cuda"):
        output = model(X)
        loss   = nn.MSELoss()(output, y)
    scaler.scale(loss).backward()
    scaler.step(opt)
    scaler.update()
else:
    # CPU: autocast and scaling are no-ops; just run normally
    output = model(X)
    loss   = nn.MSELoss()(output, y)
    loss.backward()
    opt.step()

print(f"Mixed precision step complete. Loss: {loss.item():.4f}")
Mixed precision step complete. Loss: 1.4753
SECTION 06

Metrics Tracking and Early Stopping

python
class EarlyStopping:
    """Stop training when validation loss hasn't improved for `patience` epochs."""

    def __init__(self, patience: int = 10, min_delta: float = 1e-4):
        self.patience   = patience
        self.min_delta  = min_delta
        self.best_loss  = float('inf')
        self.counter    = 0
        self.best_state = None

    def step(self, val_loss: float, model: nn.Module) -> bool:
        """Returns True if training should stop."""
        if val_loss < self.best_loss - self.min_delta:
            self.best_loss  = val_loss
            self.counter    = 0
            self.best_state = {k: v.clone() for k, v in model.state_dict().items()}
        else:
            self.counter += 1

        return self.counter >= self.patience

    def restore_best(self, model: nn.Module):
        model.load_state_dict(self.best_state)


# Usage in training loop
es = EarlyStopping(patience=10)
for epoch in range(200):
    val_loss = 0.5 * (0.99 ** epoch) + 0.01   # Simulated val loss
    if es.step(val_loss, nn.Linear(1,1)):
        print(f"Early stopping at epoch {epoch}. Best loss: {es.best_loss:.4f}")
        break

DEBUGGING TIP — Overfit a tiny batch first
Before debugging anything else, train your model on just 4-8 samples. If the loss can't reach near-zero on that, something is fundamentally broken — the model architecture, the loss function, or the data pipeline. Fix that before scaling up.

def debug_model(model, X, y, criterion):
    print("=== MODEL DEBUG ===")
    print(f"Input shape:  {X.shape}")
    print(f"Target shape: {y.shape}")

    # Check for NaN
    print(f"Any NaN in output? {output.isnan().any().item()}")
    print(f"Any NaN in input?  {X.isnan().any().item()}")

HANDS-ON PROJECT

End-to-End Tabular DS Model

Task: Predict employee churn (classification) and salary (regression)

python
torch.manual_seed(0)

# Synthetic HR dataset
N = 2000
age          = torch.randint(22, 65, (N,)).float()
experience   = (age - 22 + torch.randn(N) * 2).clamp(0)
satisfaction = torch.rand(N) * 10
hours_week   = 30 + torch.rand(N) * 30
performance  = torch.rand(N) * 5 + 1

# Target 1: Salary (regression)
salary = (experience * 2500 + age * 300 + performance * 8000
          + torch.randn(N) * 5000 + 30000)

# Target 2: Churn (binary classification)
churn_score = (-satisfaction * 0.3 + hours_week * 0.02
               - performance * 0.1 + torch.randn(N) * 0.5)
churn_label = (churn_score > churn_score.median()).long()

# Feature matrix
features = torch.stack([age, experience, satisfaction, hours_week, performance], dim=1)

# Normalise features
feat_mean = features.mean(0); feat_std = features.std(0)
features = (features - feat_mean) / feat_std

# Normalise salary target
sal_mean = salary.mean(); sal_std = salary.std()
salary_n = (salary - sal_mean) / sal_std

# Train / val split
split = int(0.8 * N)
X_tr, X_vl             = features[:split], features[split:]
y_sal_tr, y_sal_vl     = salary_n[:split], salary_n[split:]
y_churn_tr, y_churn_vl = churn_label[:split], churn_label[split:]

print(f"Train: {X_tr.shape}, Val: {X_vl.shape}")
print(f"Churn rate: {churn_label.float().mean():.2f}")

Multi-task Model: Shared Trunk + Two Heads

python
from torch.utils.data import TensorDataset, DataLoader

# DataLoaders
train_ds = TensorDataset(X_tr, y_sal_tr, y_churn_tr)
val_ds   = TensorDataset(X_vl, y_sal_vl, y_churn_vl)
train_dl = DataLoader(train_ds, batch_size=64, shuffle=True)
val_dl   = DataLoader(val_ds,   batch_size=256)

# Multi-task model: shared trunk → two heads
class HRModel(nn.Module):

    def __init__(self):
        super().__init__()
        self.trunk = nn.Sequential(
            nn.Linear(5, 64), nn.LayerNorm(64), nn.ReLU(), nn.Dropout(0.2),
            nn.Linear(64, 32), nn.LayerNorm(32), nn.ReLU(), nn.Dropout(0.1),
        )
        self.salary_head = nn.Linear(32, 1)      # Regression
        self.churn_head  = nn.Linear(32, 2)      # Classification

    def forward(self, x):
        h = self.trunk(x)
        return self.salary_head(h).squeeze(1), self.churn_head(h)


model    = HRModel()
opt      = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-3)
sched    = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=50)
mse_loss = nn.MSELoss()
ce_loss  = nn.CrossEntropyLoss()
es       = EarlyStopping(patience=8)

print(f"Model params: {sum(p.numel() for p in model.parameters()):,}")

Training with Everything: LR Schedule + Grad Clip + Early Stopping

python
# Training
for epoch in range(60):
    model.train()
    epoch_loss = 0.0
    for X_b, y_sal_b, y_churn_b in train_dl:
        opt.zero_grad()
        sal_pred, churn_pred = model(X_b)
        loss_sal   = mse_loss(sal_pred, y_sal_b)
        loss_churn = ce_loss(churn_pred, y_churn_b)
        loss       = loss_sal + 0.5 * loss_churn    # Combined loss
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        opt.step()
        epoch_loss += loss.item()

    sched.step()

    # Validation
    model.eval()
    with torch.no_grad():
        sal_p, churn_p = model(X_vl)
        val_mse        = mse_loss(sal_p, y_sal_vl).item()
        val_acc        = (churn_p.argmax(1) == y_churn_vl).float().mean().item()

    if epoch % 10 == 0:
        print(f"Epoch {epoch:3d}: train_loss={epoch_loss/len(train_dl):.4f} "
              f"val_mse={val_mse:.4f} val_churn_acc={val_acc:.3f}")

    if es.step(val_mse, model):
        print(f"\nEarly stop at epoch {epoch}"); break

es.restore_best(model)
print("\nTraining complete. Best weights restored.")
