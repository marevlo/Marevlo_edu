Moving Data to GPU

python
# The correct device management pattern
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# Tensors
x_cpu = torch.randn(1000, 512)
x_gpu = x_cpu.to(device)           # or x_cpu.cuda()
print(f"Tensor device: {x_gpu.device}")

# Models — moves ALL parameters and buffers at once
model = nn.Sequential(nn.Linear(512, 256), nn.ReLU(), nn.Linear(256, 10))
model = model.to(device)

# Training loop — move batches to device
def train_step(model, X_batch, y_batch, opt, criterion):
    X_batch = X_batch.to(device)          # Move batch to GPU
    y_batch = y_batch.to(device)
    opt.zero_grad()
    loss = criterion(model(X_batch), y_batch)
    loss.backward()
    opt.step()
    return loss.item()                    # Pull scalar back to CPU

print("Device management pattern defined.")

Avoiding Common CUDA Errors

python
# ERROR: tensors on different devices
# x_cpu = torch.randn(4, 4)
# x_gpu = torch.randn(4, 4, device='cuda')
# x_cpu + x_gpu → RuntimeError: Expected all tensors on same device

# ERROR: accumulating GPU tensor in a Python list
# losses = []
# for batch in loader:
#     loss = train_step(...)
#     losses.append(loss)        # ← This keeps GPU memory alive!
# Fix: losses.append(loss.item())   # Pull to Python float first

# ERROR: calling .numpy() on GPU tensor
# t = torch.randn(4, device='cuda')
# t.numpy() → Error
# Fix:
# t.cpu().numpy()
# or: t.detach().cpu().numpy()

print("Common CUDA error patterns documented — avoid these.")
Common CUDA error patterns documented — avoid these.
WARNING — Three CUDA mistakes that bite everyone
Mixed-device tensors throw RuntimeError. Accumulating GPU tensors in a Python list (instead of .item()) silently leaks VRAM. Calling .numpy() directly on a GPU tensor errors — chain .cpu().numpy() instead.

python
from torch.amp import autocast, GradScaler

def train_with_amp(model, loader_simulated, optimizer, criterion, device):
    """Full mixed precision training loop."""
    # Use 'cuda' scaler only when GPU available; CPU training doesn't need scaling
    device_str = str(device)
    scaler = GradScaler('cuda', enabled=(device_str == 'cuda'))
    model.train()

    for X_b, y_b in loader_simulated:
        X_b, y_b = X_b.to(device), y_b.to(device)
        optimizer.zero_grad()

        # Forward pass in float16 where safe (no-op on CPU)
        with autocast(device_type=device_str, enabled=(device_str == 'cuda')):
            output = model(X_b)
            loss   = criterion(output, y_b)

        if device_str == 'cuda':
            scaler.scale(loss).backward()
            scaler.unscale_(optimizer)
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            scaler.step(optimizer)
            scaler.update()
        else:
            # CPU path: normal backward (no scaling needed)
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()

    return loss.item()


# Simulate usage
model     = nn.Sequential(nn.Linear(32, 64), nn.ReLU(), nn.Linear(64, 4)).to(device)
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3)
criterion = nn.CrossEntropyLoss()

fake_loader = [(torch.randn(32, 32), torch.randint(0, 4, (32,))) for _ in range(5)]
loss_val    = train_with_amp(model, fake_loader, optimizer, criterion, device)

        reserved  = torch.cuda.memory_reserved() / 1e9

# Memory optimisation strategies:
# 1. Mixed precision (float16 vs float32) → 2× savings
# 2. Gradient accumulation → train with larger effective batch without OOM
# 3. Gradient checkpointing → recompute activations during backward (slower, less memory)

# Gradient checkpointing example
from torch.utils.checkpoint import checkpoint

class CheckpointedBlock(nn.Module):
    def __init__(self):
        super().__init__()
        self.layers = nn.Sequential(
            nn.Linear(128, 256), nn.ReLU(),
            nn.Linear(256, 128), nn.ReLU()
        )

    def forward(self, x):
        # Checkpointed: don't store activations; recompute in backward
        return checkpoint(self.layers, x, use_reentrant=False)

Multi-GPU: DDP Basics

DDP (DistributedDataParallel) is the standard way to train on multiple GPUs. Each GPU runs a full copy of the model; gradients are synchronised after each backward pass.

DDP Gradient All-Reduce Simulator
N GPUs each compute gradients on their local data shard, then synchronise. The all-reduce averages gradients across all GPUs — every GPU ends up with identical updated weights.

# DDP requires a proper launch script (torchrun or mp.spawn)
# Here is the template — run as a script, not a notebook cell

DDP_TEMPLATE = '''
import torch
import torch.nn as nn
import torch.distributed as dist
from torch.nn.parallel import DistributedDataParallel as DDP
from torch.utils.data import DataLoader, DistributedSampler

def setup(rank, world_size):
    dist.init_process_group("nccl", rank=rank, world_size=world_size)
    torch.cuda.set_device(rank)

def cleanup():
    dist.destroy_process_group()

def train(rank, world_size, dataset):
    setup(rank, world_size)
    device = torch.device(f"cuda:{rank}")

    # Each GPU gets a non-overlapping shard of the dataset
    sampler    = DistributedSampler(dataset, num_replicas=world_size, rank=rank)
    dataloader = DataLoader(dataset, batch_size=64, sampler=sampler)

    model     = MyModel().to(device)
    model     = DDP(model, device_ids=[rank])               # Wrap with DDP
    optimizer = torch.optim.AdamW(model.parameters())
    criterion = nn.CrossEntropyLoss()

    for epoch in range(10):
        sampler.set_epoch(epoch)                             # Important: reshuffle each epoch
        for X_b, y_b in dataloader:
            X_b, y_b = X_b.to(device), y_b.to(device)
            optimizer.zero_grad()
            loss = criterion(model(X_b), y_b)
            loss.backward()                                 # Gradients synced automatically
            optimizer.step()

        if rank == 0:                                   # Only rank-0 saves/logs
            print(f"Epoch {epoch}: loss={loss.item():.4f}")

    cleanup()

if __name__ == "__main__":
    import torch.multiprocessing as mp
    world_size = torch.cuda.device_count()
    mp.spawn(train, args=(world_size, dataset), nprocs=world_size)
'''

print("DDP template written.")
print("Launch with: torchrun --nproc_per_node=NUM_GPUS train.py")
DDP template written. Launch with: torchrun --nproc_per_node=NUM_GPUS train.py
TIP — DDP gotchas to remember
Always call sampler.set_epoch(epoch) — without it, every epoch sees the same shuffle. Only rank 0 should save checkpoints and log to disk. Use NCCL backend on Nvidia GPUs (Gloo for CPU-only). When in doubt: torchrun --nproc_per_node=NUM_GPUS train.py.

    record_shapes=True,
    profile_memory=True,

# Simple throughput benchmark
import time

def benchmark_throughput(model, batch_size=256, n_batches=100, input_dim=256):
    model.eval()
    x = torch.randn(batch_size, input_dim).to(device)

    # Warmup
    with torch.no_grad():
        for _ in range(5): model(x)

    if torch.cuda.is_available():
        torch.cuda.synchronize()

    start = time.time()
    with torch.no_grad():
        for _ in range(n_batches): model(x)

    if torch.cuda.is_available():
        torch.cuda.synchronize()

    elapsed = time.time() - start
    samples_per_sec = (n_batches * batch_size) / elapsed
    print(f"Throughput: {samples_per_sec:,.0f} samples/sec "
          f"({elapsed:.3f}s for {n_batches} batches of {batch_size})")
