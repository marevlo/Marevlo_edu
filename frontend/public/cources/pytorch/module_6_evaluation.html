SECTION 01

Train / Val / Test Patterns

The Three Roles

SplitPurposeRules
TrainFit model parametersCan see as many times as you want
ValidationTune hyperparameters, select modelsSee during development, but don't fit on it
TestFinal, unbiased performance estimateLook at it exactly once
IMPORTANT — The test set is sacred
The moment you tune anything based on test set results, it becomes a second validation set and your reported performance is optimistic. Report test performance only once — at the very end.

class SimpleClassifier(nn.Module):
    def __init__(self, in_dim=10):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, 64), nn.ReLU(), nn.Dropout(0.2),
            nn.Linear(64, 32),     nn.ReLU(),
            nn.Linear(32, 1)       # Binary: single logit
        )
    def forward(self, x):
        return self.net(x).squeeze(1)


def evaluate_binary(model: nn.Module, X: torch.Tensor, y: torch.Tensor,
                    threshold: float = 0.5) -> dict:
    """
    Full binary classification evaluation.
    Returns: loss, accuracy, precision, recall, f1, auroc, probs, preds
    """
    model.eval()
    with torch.no_grad():
        logits = model(X)
        probs  = torch.sigmoid(logits)
        preds  = (probs >= threshold).long()
        loss   = nn.BCEWithLogitsLoss()(logits, y.float()).item()

    # Confusion matrix components
    tp = ((preds == 1) & (y == 1)).sum().float()
    fp = ((preds == 1) & (y == 0)).sum().float()
    fn = ((preds == 0) & (y == 1)).sum().float()
    tn = ((preds == 0) & (y == 0)).sum().float()

    precision = (tp / (tp + fp + 1e-8)).item()
    recall    = (tp / (tp + fn + 1e-8)).item()
    f1        = (2 * precision * recall / (precision + recall + 1e-8))
    accuracy  = ((tp + tn) / len(y)).item()

    return {
        'loss':      loss,
        'accuracy':  accuracy,
        'precision': precision,
        'recall':    recall,
        'f1':        f1,
        'probs':     probs,
        'preds':     preds,
        'tp': tp.item(), 'fp': fp.item(),
        'fn': fn.item(), 'tn': tn.item()
    }

# Train a model quickly so we can evaluate it
model = SimpleClassifier()
opt   = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-3)

# Class-weighted loss for imbalance
pos_weight = torch.tensor([(y_train == 0).sum() / (y_train == 1).sum()])
criterion  = nn.BCEWithLogitsLoss(pos_weight=pos_weight)

for epoch in range(80):
    model.train()
    opt.zero_grad()
    logits = model(X_train)
    loss   = criterion(logits, y_train.float())
    loss.backward()
    opt.step()

print("Training complete.")
metrics = evaluate_binary(model, X_val, y_val, threshold=0.5)
print(f"\nVal @ threshold=0.5:")
print(f"  Accuracy:  {metrics['accuracy']:.3f}")
print(f"  Precision: {metrics['precision']:.3f}")
print(f"  Recall:    {metrics['recall']:.3f}")

SECTION 03

Calibration, Thresholding, PR / ROC

Why Threshold Matters

The default threshold of 0.5 is rarely optimal for imbalanced tasks. Shift it to trade Precision for Recall depending on business cost.

  • High Recall, Low Precision: catch most fraud, but flag many false positives (costly for customers).
  • High Precision, Low Recall: only flag high-confidence fraud, miss some cases (costly for losses).

SECTION 04

Cross-Validation in PyTorch

When to Use It

Use k-fold CV when:

  • Your dataset is small (< 5,000 samples) and a single val split is noisy.
  • You want a reliable estimate of generalisation performance.
  • You are comparing multiple models/architectures.

Do not use it on time-series (use time-series CV instead).

from torch.utils.data import Subset

def kfold_indices(n: int, k: int = 5, seed: int = 42):
    """Generate k-fold train/val index pairs."""
    rng     = np.random.default_rng(seed)
    indices = rng.permutation(n)
    folds   = np.array_split(indices, k)
    for i in range(k):
        val_idx   = folds[i]
        train_idx = np.concatenate([folds[j] for j in range(k) if j != i])
        yield train_idx.tolist(), val_idx.tolist()


def train_and_evaluate_fold(X, y, train_idx, val_idx, n_epochs=50):
    X_tr, y_tr = X[train_idx], y[train_idx]
    X_vl, y_vl = X[val_idx],   y[val_idx]

    m   = SimpleClassifier(X.shape[1])
    opt = torch.optim.AdamW(m.parameters(), lr=1e-3)
    pw  = torch.tensor([(y_tr == 0).sum() / (y_tr == 1).sum().clamp(1)])
    crit = nn.BCEWithLogitsLoss(pos_weight=pw)

    for _ in range(n_epochs):
        m.train(); opt.zero_grad()
        loss = crit(m(X_tr), y_tr.float())
        loss.backward(); opt.step()

    res = evaluate_binary(m, X_vl, y_vl, threshold=best_t)
    return res['f1']

def bootstrap_ci(y_true: np.ndarray, y_prob: np.ndarray,
                 metric_fn, n_bootstrap: int = 1000,
                 confidence: float = 0.95, threshold: float = 0.5):
    """
    Bootstrap confidence interval for any binary classification metric.
    """
    scores = []
    n = len(y_true)
    rng = np.random.default_rng(42)

    for _ in range(n_bootstrap):
        idx = rng.integers(0, n, size=n)   # Sample with replacement
        scores.append(metric_fn(y_true[idx], y_prob[idx], threshold))

    alpha = (1 - confidence) / 2
    lower = np.percentile(scores, alpha * 100)
    upper = np.percentile(scores, (1 - alpha) * 100)
    return np.mean(scores), lower, upper


def f1_fn(y_true, y_prob, threshold):
    preds = (y_prob >= threshold).astype(int)
    tp = ((preds == 1) & (y_true == 1)).sum()
    fp = ((preds == 1) & (y_true == 0)).sum()
    fn = ((preds == 0) & (y_true == 1)).sum()
    p  = tp / (tp + fp + 1e-8)
    r  = tp / (tp + fn + 1e-8)
    return 2 * p * r / (p + r + 1e-8)

Error Analysis Workflow

The Systematic Approach When Results Are Bad

def error_analysis(model, X, y, feature_names=None, threshold=0.5):
    """
    Full error analysis: find where and why the model fails.
    """
    model.eval()
    with torch.no_grad():
        probs = torch.sigmoid(model(X)).numpy()
    preds = (probs >= threshold).astype(int)
    y_np  = y.numpy()
    X_np  = X.numpy()

    fp_mask = (preds == 1) & (y_np == 0)   # False Positives: predicted fraud, was normal
    fn_mask = (preds == 0) & (y_np == 1)   # False Negatives: missed fraud

    print("=== Error Analysis ===")
    print(f"False Positives: {fp_mask.sum()}  (predicted positive, actually negative)")
    print(f"False Negatives: {fn_mask.sum()}  (predicted negative, actually positive)")

    if feature_names:
        print("\nFalse Positive mean features vs overall mean:")
        for i, name in enumerate(feature_names):
            overall_mean = X_np[:, i].mean()
            fp_mean      = X_np[fp_mask, i].mean() if fp_mask.sum() > 0 else float('nan')
            print(f"  {name:12s}: overall={overall_mean:.3f}  FP={fp_mean:.3f}")

    # Confidence distribution of errors
    fp_conf = probs[fp_mask]
    fn_conf = probs[fn_mask]
    if len(fp_conf) > 0:
        print(f"\nFP confidence: mean={fp_conf.mean():.3f}, max={fp_conf.max():.3f}")
    if len(fn_conf) > 0:
        print(f"FN confidence: mean={fn_conf.mean():.3f}, min={fn_conf.min():.3f}")


feat_names = [f"feature_{i}" for i in range(10)]
error_analysis(model, X_val, y_val, feature_names=feat_names, threshold=best_t)

HANDS-ON PROJECT

Imbalanced Classification Pipeline

Full pipeline from data to production-ready threshold

python
class FraudDetectionPipeline:

    def __init__(self, in_dim: int, hidden: int = 64):
        self.model     = SimpleClassifier(in_dim)
        self.opt       = torch.optim.AdamW(self.model.parameters(), lr=1e-3)
        self.threshold = 0.5
        self.history   = {'train_loss': [], 'val_f1': []}

    def fit(self, X_train, y_train, X_val, y_val, n_epochs=100, patience=15):
        pw    = torch.tensor([(y_train == 0).sum() / (y_train == 1).sum().clamp(1)])
        crit  = nn.BCEWithLogitsLoss(pos_weight=pw)
        sched = torch.optim.lr_scheduler.CosineAnnealingLR(self.opt, T_max=n_epochs)

        best_f1, best_state, counter = 0.0, None, 0

        for epoch in range(n_epochs):
            self.model.train()
            self.opt.zero_grad()
            loss = crit(self.model(X_train), y_train.float())
            loss.backward(); self.opt.step(); sched.step()

            res = evaluate_binary(self.model, X_val, y_val, self.threshold)
            self.history['train_loss'].append(loss.item())
            self.history['val_f1'].append(res['f1'])

            if res['f1'] > best_f1:
                best_f1    = res['f1']
                best_state = {k: v.clone() for k, v in self.model.state_dict().items()}
                counter    = 0
            else:
                counter += 1
                if counter >= patience:
                    print(f"Early stop at epoch {epoch}")
                    break

        self.model.load_state_dict(best_state)
        self._tune_threshold(X_val, y_val)
        return self

    def _tune_threshold(self, X_val, y_val):
        """Find the threshold maximising F1 on validation set."""
        self.model.eval()
        with torch.no_grad():
            probs = torch.sigmoid(self.model(X_val)).numpy()

        thresholds = np.linspace(0.05, 0.95, 91)
        best_f1, best_t = 0, 0.5
        for t in thresholds:
            preds = (probs >= t).astype(int)
            tp = ((preds == 1) & (y_val.numpy() == 1)).sum()
            fp = ((preds == 1) & (y_val.numpy() == 0)).sum()
            fn = ((preds == 0) & (y_val.numpy() == 1)).sum()
            f1 = 2*tp / (2*tp + fp + fn + 1e-8)
            if f1 > best_f1:
                best_f1, best_t = f1, t

        self.threshold = best_t
        print(f"Tuned threshold: {self.threshold:.2f}  (Val F1={best_f1:.3f})")

    def evaluate(self, X, y, label="Test"):
        res = evaluate_binary(self.model, X, y, self.threshold)
        print(f"\n{label} Results (threshold={self.threshold:.2f}):")
        for k, v in res.items():
            if k not in ('probs', 'preds'):
                print(f"  {k:12s}: {v:.4f}" if isinstance(v, float) else f"  {k:12s}: {v}")
        return res

    def predict_proba(self, X):
        self.model.eval()
        with torch.no_grad():
            return torch.sigmoid(self.model(X))


pipeline = FraudDetectionPipeline(in_dim=10)
