SECTION 01

Dataset vs IterableDataset

The Two Modes of Data Loading

DatasetIterableDataset
Access patternRandom access via __getitem__(i)Sequential stream
Must know length?Yes (__len__ required)No
ShufflingFree — DataLoader handles itMust implement manually
Best forFiles on disk, databases, in-memory dataStreaming logs, large files, network data

Real-World Analogy

Dataset is like a library — you can walk to any shelf and grab book number 247.
IterableDataset is like a conveyor belt — items come to you in order, and you process them as they arrive.

Which Dataset Type Should You Use?
Pick a data scenario and see which Dataset subclass fits, with the reasoning.
Dataset (map-style)
  • Random access via index
  • Free shuffling by DataLoader
  • Easy train / val / test split
  • Requires __len__
IterableDataset (stream)
  • Lazy, line-by-line iteration
  • No memory cost for unread items
  • Manual worker sharding
  • No built-in shuffling

Map-Style Dataset

python
import torch
from torch.utils.data import Dataset, DataLoader
import numpy as np

class TabularDataset(Dataset):
    """
    A reusable Dataset for tabular data stored as tensors or numpy arrays.
    Applies optional per-feature normalisation.
    """

    def __init__(self, X: torch.Tensor, y: torch.Tensor, transform=None):
        assert len(X) == len(y), "X and y must have the same number of samples"
        self.X         = X.float()
        self.y         = y
        self.transform = transform

    def __len__(self) -> int:
        return len(self.X)

    def __getitem__(self, idx: int):
        x = self.X[idx]
        if self.transform:
            x = self.transform(x)
        return x, self.y[idx]


# Usage
torch.manual_seed(42)
X_raw = torch.randn(1000, 8)
y_raw = (X_raw[:, 0] + X_raw[:, 1] > 0).long()

dataset = TabularDataset(X_raw, y_raw)
print(f"Dataset length: {len(dataset)}")
print(f"Sample 0: X shape={dataset[0][0].shape}, y={dataset[0][1].item()}")

Iterable Dataset — For Large Log Files

python
from torch.utils.data import IterableDataset
import itertools

class LogFileDataset(IterableDataset):
    """
    Streams log lines from one or more files without loading everything into RAM.
    Yields (token_ids, label) tuples.
    Each line is a space-separated sequence of token IDs followed by a label.
    """

    def __init__(self, filepaths: list, max_seq_len: int = 64):
        self.filepaths   = filepaths
        self.max_seq_len = max_seq_len

    def _parse_line(self, line: str):
        tokens = list(map(int, line.strip().split()))
        label  = tokens[-1]
        seq    = tokens[:-1][:self.max_seq_len]         # Truncate to max_seq_len
        return torch.tensor(seq, dtype=torch.long), label

    def __iter__(self):
        worker_info = torch.utils.data.get_worker_info()
        files = self.filepaths

        # Distribute files across workers if multi-worker loading
        if worker_info is not None:
            files = [f for i, f in enumerate(files)
                     if i % worker_info.num_workers == worker_info.id]

        for filepath in files:
            with open(filepath, 'r') as f:
                for line in f:
                    if line.strip():
                        yield self._parse_line(line)

Collate Functions — Handling Variable-Length Sequences

The Problem

By default, DataLoader tries to stack samples into a single tensor. This works when all samples have the same shape. But log sequences, text, or time-series windows often have variable length — stacking fails.

Padding Collate Function

python
from torch.nn.utils.rnn import pad_sequence

def collate_variable_length(batch):
    """
    Collate a batch of (sequence, label) tuples with variable-length sequences.
    Pads sequences to the longest in the batch.

    Returns:
        sequences: (B, T_max) padded tensor
        lengths:   (B,)       original lengths (needed by LSTM)
        labels:    (B,)       labels
    """
    sequences, labels = zip(*batch)

    lengths = torch.tensor([len(s) for s in sequences])
    padded  = pad_sequence(sequences, batch_first=True, padding_value=0)
    labels  = torch.tensor(labels)

    return padded, lengths, labels


# Test it
batch = [
    (torch.tensor([1, 2, 3]),     0),
    (torch.tensor([4, 5]),        1),
    (torch.tensor([6, 7, 8, 9]), 0),
]

padded, lengths, labels = collate_variable_length(batch)
print("Padded sequences:\n", padded)
print("Original lengths:", lengths)
print("Labels:          ", labels)
Padded sequences: tensor([[1, 2, 3, 0], [4, 5, 0, 0], [6, 7, 8, 9]]) Original lengths: tensor([3, 2, 4]) Labels: tensor([0, 1, 0])

Truncate + Pad Collate for Fixed Bucket Size

python
def make_collate_fn(max_len: int = 32, pad_value: int = 0):
    """Returns a collate_fn that truncates to max_len then pads."""

    def collate(batch):
        sequences, labels = zip(*batch)
        truncated = [s[:max_len] for s in sequences]
        lengths   = torch.tensor([len(s) for s in truncated])
        padded    = pad_sequence(truncated, batch_first=True, padding_value=pad_value)
        return padded, lengths, torch.tensor(labels)

    return collate


# Use in DataLoader
collate_fn = make_collate_fn(max_len=16)

SECTION 04

Streaming Large Datasets

The Memory Problem

A large log file (100GB+) cannot be loaded into RAM. Solutions:

  1. IterableDataset — stream line by line.
  2. Memory-mapped files — np.memmap or torch.frombuffer for binary data.
  3. Chunked HDF5/Parquet — load one shard at a time.

python
import struct

# Memory-mapped binary dataset
class BinarySequenceDataset(Dataset):
    """
    Reads fixed-length sequences from a binary file using memory mapping.
    Format: each record = [n_tokens int32][tokens int32 * seq_len][label int32]
    """

    def __init__(self, filepath: str, seq_len: int, dtype=np.float32):
        self.seq_len   = seq_len
        self.data      = np.memmap(filepath, dtype=dtype, mode='r')
        self.n_samples = len(self.data) // (seq_len + 1)

    def __len__(self):
        return self.n_samples

    def __getitem__(self, idx):
        start = idx * (self.seq_len + 1)
        row   = self.data[start : start + self.seq_len + 1]
        x     = torch.tensor(row[:-1], dtype=torch.float32)
        y     = int(row[-1])
        return x, y

# Correct normalisation — fit ONLY on train, apply to all
X_full = torch.randn(1000, 8)
train_X = X_full[:700]
val_X   = X_full[700:850]
test_X  = X_full[850:]

# Fit stats on TRAIN only
mean = train_X.mean(dim=0)
std  = train_X.std(dim=0)
std[std < 1e-6] = 1.0

# Apply to all splits
train_X_norm = (train_X - mean) / std
val_X_norm   = (val_X   - mean) / std               # ← use train stats, not val stats
test_X_norm  = (test_X  - mean) / std               # ← use train stats, not test stats

IMPORTANT — Data leakage checklist
Never compute mean/std/quantiles on the full dataset before splitting. Never include future information in time-series features. Never let validation samples influence training augmentation parameters.

Log Dataset Loader for AIOps

Build a complete log sequence loader with windowing and anomaly labels

python
import hashlib

# Simulate raw log data
# In real AIOps, this would be parsed from syslog/ELK/Splunk
# Here: each "event" is a log template ID (0-99), with timestamps

torch.manual_seed(99)
np.random.seed(99)

def generate_log_stream(n_events: int = 5000, vocab_size: int = 50,
                        anomaly_rate: float = 0.05):
    """
    Simulate a log stream.
    Normal: Markov-ish transitions between common templates.
    Anomaly: random rare templates injected.
    """
    events    = []
    labels    = []   # 1 = anomaly window, 0 = normal window
    templates = np.arange(vocab_size)

    # Build a transition matrix (normal behaviour)
    trans = np.random.dirichlet(np.ones(vocab_size) * 0.3, size=vocab_size)

    current = 0
    for i in range(n_events):
        is_anomaly = np.random.rand() < anomaly_rate
        if is_anomaly:
            # Rare template — high ID
            event = np.random.randint(40, vocab_size)
        else:
            event = np.random.choice(templates, p=trans[current])
        events.append(event)
        labels.append(int(is_anomaly))
        current = event

    return np.array(events), np.array(labels)


log_events, event_labels = generate_log_stream(5000)
print(f"Log stream: {len(log_events)} events")
print(f"Anomaly rate: {event_labels.mean():.3f}")

Sliding-Window Dataset

python
class LogWindowDataset(Dataset):
    """
    Sliding-window dataset from a log event stream.
    Each sample = window of `window_size` events → label = 1 if any anomaly in window.
    """

    def __init__(self, events: np.ndarray, labels: np.ndarray,
                 window_size: int = 20, stride: int = 1):
        self.window_size = window_size
        self.stride      = stride

        windows    = []
        win_labels = []

        for start in range(0, len(events) - window_size, stride):
            end = start + window_size
            windows.append(events[start:end])
            win_labels.append(int(labels[start:end].any()))   # 1 if ANY anomaly

        self.X = torch.tensor(np.array(windows), dtype=torch.long)
        self.y = torch.tensor(win_labels, dtype=torch.long)

    def __len__(self):
        return len(self.X)

    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]


log_dataset = LogWindowDataset(log_events, event_labels, window_size=20, stride=5)

print(f"Window dataset: {len(log_dataset)} windows")
print(f"X shape: {log_dataset.X.shape}")
print(f"Anomaly window rate: {log_dataset.y.float().mean():.3f}")

# Check class imbalance
pos = log_dataset.y.sum().item()
neg = len(log_dataset) - pos
print(f"Normal windows: {neg}, Anomaly windows: {pos}")

Weighted Sampler for Class Imbalance

python
# Temporal split (no shuffle for sequence data)
n      = len(log_dataset)
split1 = int(0.7 * n)
split2 = int(0.85 * n)

train_ids = list(range(0, split1))
val_ids   = list(range(split1, split2))
test_ids  = list(range(split2, n))

from torch.utils.data import Subset

train_log = Subset(log_dataset, train_ids)
val_log   = Subset(log_dataset, val_ids)
test_log  = Subset(log_dataset, test_ids)

# Weighted sampler to handle class imbalance
from torch.utils.data import WeightedRandomSampler

train_labels = log_dataset.y[train_ids]
class_counts = torch.bincount(train_labels)
weights      = 1.0 / class_counts.float()
sample_weights = weights[train_labels]

sampler = WeightedRandomSampler(
    weights=sample_weights,
    num_samples=len(train_log),
    replacement=True
)

train_dl_log = DataLoader(train_log, batch_size=64, sampler=sampler)
val_dl_log   = DataLoader(val_log,   batch_size=256, shuffle=False)

X_b, y_b = next(iter(train_dl_log))
print(f"\nBatch from log DataLoader:")
print(f"  X shape: {X_b.shape}")
print(f"  y distribution in batch: {y_b.bincount()}")
